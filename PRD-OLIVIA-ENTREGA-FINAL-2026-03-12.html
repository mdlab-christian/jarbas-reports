5.2 Hybrid Mode — Smart Blocks sem n8n

// /src/hooks/olivia/useSmartBlocksHybrid.ts
export type GeracaoMode = 'n8n' | 'edge' | 'direct';

async function detectMode(): Promise<GeracaoMode> {
  const cached = localStorage.getItem('olivia_n8n_health');
  if (cached && Date.now() - JSON.parse(cached).ts < 300_000) {
    return JSON.parse(cached).mode;
  }
  try {
    const res = await fetch('https://primary-production-e209.up.railway.app/healthz', { signal: AbortSignal.timeout(3000) });
    const mode = res.ok ? 'n8n' : 'edge';
    localStorage.setItem('olivia_n8n_health', JSON.stringify({ mode, ts: Date.now() }));
    return mode;
  } catch { return 'edge'; }
}

// Fallback order: n8n → Supabase Edge Function → Anthropic direct (server-side)
