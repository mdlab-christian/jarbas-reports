```sql
-- Fix HP-01: get_homepage_ranking referencia p.responsavel_id que nao existe
-- Coluna correta em processos eh advogado_id
CREATE OR REPLACE FUNCTION get_homepage_ranking(
  p_user_id UUID,
  p_org_id  UUID
)
RETURNS TABLE(
  user_id    UUID,
  user_nome  TEXT,
  avatar_url TEXT,
  perfil     TEXT,
  score      NUMERIC
)
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public
AS $$
DECLARE
  v_acesso_gestor BOOLEAN;
BEGIN
  SELECT acesso_gestor INTO v_acesso_gestor
  FROM users WHERE id = p_user_id;

  IF NOT COALESCE(v_acesso_gestor, false) THEN
    RAISE EXCEPTION 'P0001: acesso_gestor negado' USING HINT = 'ACCESS_DENIED';
  END IF;

  RETURN QUERY
  SELECT
    u.id,
    u.nome,
    u.avatar_url,
    u.perfil,
    COALESCE(
      (COUNT(a.id) FILTER (
        WHERE a.status = 'concluida'
          AND a.concluida_em >= DATE_TRUNC('month', now())
      ) * 2
      + COUNT(DISTINCT p.id) FILTER (WHERE p.status = 'ativo'))::NUMERIC,
      0
    ) AS score
  FROM users u
  LEFT JOIN atividades a ON a.responsavel_id = u.id
    AND a.organizacao_id = p_org_id
    AND a.deleted_at IS NULL
  LEFT JOIN processos p ON p.advogado_id = u.id
    AND p.organizacao_id = p_org_id
    AND p.deleted_at IS NULL
  WHERE u.organizacao_id = p_org_id
    AND u.ativo = true
  GROUP BY u.id, u.nome, u.avatar_url, u.perfil
  ORDER BY score DESC
  LIMIT 10;
END;
$$;
```

P1 HP-04: Corrigir RPC taxa_conversao > 100% Independente

Arquivo: ~/beta-mdflow/supabase/migrations/20260306_C5_rpcs_p1.sql (linhas 767-777)

Causa raiz: A formula calcula COUNT(DISTINCT p.id) / NULLIF(COUNT(DISTINCT c.id), 0) * 100. O LEFT JOIN de clientes para processos pode gerar multiplos processos por cliente, fazendo o numerador ser maior que o denominador. Resultado: 185.7% (impossivel).

Solucao: A taxa deve ser: clientes COM pelo menos 1 processo / total de clientes * 100. Usar subquery ou CASE para cap em 100%.

Corrigir o calculo de taxa_conversao na RPC `get_homepage_gestor_comercial`.

## Problema
A RPC em `~/beta-mdflow/supabase/migrations/20260306_C5_rpcs_p1.sql` (linhas 767-777) calcula taxa_conversao como:
```sql
COUNT(DISTINCT p.id) / NULLIF(COUNT(DISTINCT c.id), 0) * 100
```
Isso retorna 185.7% porque conta TODOS os processos (varios por cliente), nao clientes com processo.

```sql
-- Fix HP-04: taxa_conversao retornava > 100%
-- A formula contava processos em vez de clientes com processo
CREATE OR REPLACE FUNCTION get_homepage_gestor_comercial(
  p_user_id UUID,
  p_org_id  UUID,
  p_periodo TEXT DEFAULT '30d'
)
RETURNS JSONB
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public
AS $$
DECLARE
  v_inicio TIMESTAMPTZ;
  v_acesso_gestor BOOLEAN;
BEGIN
  SELECT acesso_gestor INTO v_acesso_gestor
  FROM users WHERE id = p_user_id;

  IF NOT COALESCE(v_acesso_gestor, false) THEN
    RAISE EXCEPTION 'P0001: acesso_gestor negado' USING HINT = 'ACCESS_DENIED';
  END IF;

  v_inicio := CASE p_periodo
    WHEN 'hoje' THEN CURRENT_DATE::TIMESTAMPTZ
    WHEN '7d'   THEN now() - INTERVAL '7 days'
    ELSE              now() - INTERVAL '30 days'
  END;

  RETURN jsonb_build_object(
    'leads_novos', (
      SELECT COUNT(*) FROM leads
      WHERE organizacao_id = p_org_id
        AND created_at >= v_inicio
        AND deleted_at IS NULL
    ),
    'leads_em_atendimento', (
      SELECT COUNT(*) FROM leads
      WHERE organizacao_id = p_org_id
        AND coluna_crm = 'em_atendimento'
        AND deleted_at IS NULL
    ),
    'conversoes', (
      SELECT COUNT(*) FROM processos
      WHERE organizacao_id = p_org_id
        AND created_at >= v_inicio
        AND deleted_at IS NULL
    ),
    'taxa_conversao', (
      SELECT COALESCE(ROUND(
        COUNT(DISTINCT c.id) FILTER (WHERE EXISTS (
          SELECT 1 FROM processos p
          WHERE p.cliente_id = c.id
            AND p.created_at >= v_inicio
            AND p.deleted_at IS NULL
        ))::NUMERIC
        / NULLIF(COUNT(DISTINCT c.id), 0) * 100
      , 1), 0)
      FROM clientes c
      WHERE c.organizacao_id = p_org_id
        AND c.created_at >= v_inicio
        AND c.deleted_at IS NULL
    ),
    'indicacoes', (
      SELECT COUNT(*) FROM indicacoes
      WHERE organizacao_id = p_org_id
        AND created_at >= v_inicio
    ),
    'indicacoes_convertidas', (
      SELECT COUNT(*) FROM indicacoes
      WHERE organizacao_id = p_org_id
        AND status IN ('convertida','paga')
        AND created_at >= v_inicio
    )
  );
END;
$$;
```
