## PASSO 3 — Implementar a propagação
Após o loop/INSERT de criação de processos na RPC, adicionar o bloco de propagação:

```sql
-- Buscar documentos gerais do lead
-- (upload central: restricao_id IS NULL, categoria='documento_lead')
WITH docs_gerais AS (
  SELECT id, nome, arquivo_url, tipo_documento, organizacao_id, tipo, status
  FROM processos_documentos
  WHERE lead_id = p_lead_id
    AND restricao_id IS NULL
    AND categoria = 'documento_lead'
    AND deleted_at IS NULL
),
-- Buscar processos recém-criados (ou todos do lead ainda sem docs gerais vinculados)
novos_processos AS (
  SELECT p.id as processo_id, p.restricao_id, p.organizacao_id
  FROM processos p
  WHERE p.lead_id = p_lead_id
    AND p.deleted_at IS NULL
    -- Só processos que ainda não têm cópia desses docs gerais
    AND NOT EXISTS (
      SELECT 1 FROM processos_documentos pd
      WHERE pd.processo_id = p.id
        AND pd.categoria = 'documento_lead'
        AND pd.deleted_at IS NULL
    )
)
INSERT INTO processos_documentos (
  processo_id, restricao_id, lead_id, organizacao_id,
  nome, arquivo_url, tipo_documento, categoria,
  tipo, status, entidade_tipo
)
SELECT
  np.processo_id,
  np.restricao_id,
  p_lead_id,
  np.organizacao_id,
  dg.nome,
  dg.arquivo_url,
  dg.tipo_documento,
  'documento_lead',
  dg.tipo,
  dg.status,
  'processo'
FROM novos_processos np
CROSS JOIN docs_gerais dg;
```
