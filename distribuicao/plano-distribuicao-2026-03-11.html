🏗️Como o JARBAS Distribui — Para Gustavo
ℹ️
Contexto para Gustavo O JARBAS é um sistema Node.js rodando no Mac Mini M4 em paralelo ao MdFlow. Ele não usa banco de dados próprio — tudo passa pelo Supabase (jarbas_tarefas). O front-end cria tarefas no banco e o JARBAS as consome em background, executa navegação real no EPROC via Playwright e salva o resultado de volta no Supabase.

CREATE TABLE eproc_erros_aprendidos (
  id          UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  hash_erro   TEXT NOT NULL UNIQUE, -- hash da mensagem/screenshot
  erro_tipo   TEXT NOT NULL,        -- ex: 'CAMPO_NAO_ENCONTRADO'
  etapa       TEXT,                 -- etapa1..etapa5
  tribunal    TEXT,                 -- TJRS|TJSC|TJSP
  descricao   TEXT,                 -- o que a IA viu
  solucao_js  TEXT,                 -- código JS a executar
  solucao_desc TEXT,                -- explicação da solução
  tentativas  INT DEFAULT 0,
  sucessos    INT DEFAULT 0,
  taxa_sucesso FLOAT GENERATED ALWAYS AS
    (CASE WHEN tentativas > 0 THEN sucessos::float/tentativas ELSE 0 END) STORED,
  criado_em   TIMESTAMPTZ DEFAULT now(),
  atualizado_em TIMESTAMPTZ DEFAULT now()
);

// Adicionar ao errorResult handler — ativar Vision Agent em erros de navegação
async function visionAgentRecovery(page, error, etapa, tribunal, payload) {
  const MAX_RECOVERY_ATTEMPTS = 3;
  
  // 1. Screenshot da tela atual
  const screenshot = await screenshotEtapa(page, 'vision_recovery');
  
  // 2. Buscar erros conhecidos similares no banco
  const errosConhecidos = await buscarErrosConhecidos(error.message, etapa, tribunal);
  
  // 3. Se já existe solução com alta confiança, usar direto
  const solucaoRapida = errosConhecidos.find(e => e.taxa_sucesso > 0.8);
  if (solucaoRapida) {
    log('Vision Agent: Usando solução conhecida:', solucaoRapida.solucao_desc);
    await page.evaluate(solucaoRapida.solucao_js);
    await registrarTentativa(solucaoRapida.id, true);
    return { resolvido: true, solucao: solucaoRapida };
  }
  
  // 4. Chamar Claude Vision para analisar
  const analise = await claudeVisionAnalisar(screenshot, error, etapa, tribunal, errosConhecidos);
  
  if (analise.confianca < 0.6) {
    // Baixa confiança — perguntar ao Gustavo (via Telegram)
    await notificarGustavo(page, error, analise, screenshot);
    return { resolvido: false, aguardando_gustavo: true };
  }
  
  // 5. Executar ação sugerida
  try {
    await page.evaluate(analise.codigo_js);
    await page.waitForTimeout(2000);
    
    // 6. Registrar solução aprendida
    if (analise.aprender) {
      await registrarSolucaoAprendida(error.message, etapa, tribunal, analise);
    }
    return { resolvido: true, solucao: analise };
  } catch (e) {
    log('Vision Agent: Ação falhou:', e.message);
    return { resolvido: false, erro_recovery: e.message };
  }
}
