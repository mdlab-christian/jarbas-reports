Princípio Anti-Alucinação (modo Agente):

Antes de gerar a petição final, o Agente SEMPRE executa internamente o fluxo Smart Blocks (tags → blocos → jurisprudências → substituição placeholders) para gerar uma petição-base mecânica. Essa petição-base é REFERÊNCIA — o Agente aprimora com análise IA das provas, impugnações específicas e argumentos contextualizados. IA nunca inventa: sempre segue modelo como esqueleto + dados do backend.

🔧 Helper require('https') — validado E2E (usar em TODOS os Code nodes)

const https = require('https');
function httpRequest(options) {
  return new Promise((resolve, reject) => {
    const { method, hostname, path, headers, body } = options;
    const req = https.request({ method: method || 'GET', hostname, path, headers, port: 443 }, (res) => {
      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => {
        try { resolve({ status: res.statusCode, data: JSON.parse(data) }); }
        catch(e) { resolve({ status: res.statusCode, data }); }
      });
    });
    req.on('error', reject);
    req.setTimeout(25000, () => { req.destroy(); reject(new Error('Timeout')); });
    if (body) req.write(typeof body === 'string' ? body : JSON.stringify(body));
    req.end();
  });
}
