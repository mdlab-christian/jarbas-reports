1. Criar função utilitária `src/pages/olivia/lib/validatePlaceholders.ts`:

export interface PlaceholderValidation {
  valid: boolean;
  missing: Array<{ placeholder: string; label: string }>;
}

// Extrai {{PLACEHOLDER}} do template
export function extractPlaceholders(template: string): string[] {
  return [...template.matchAll(/\{\{([A-Z_]+)\}\}/g)].map(m => m[1]);
}

// Mapa de placeholders → campos do processo
const PLACEHOLDER_LABEL: Record = {
  NOME_CLIENTE: 'Nome do Cliente',
  CPF_CLIENTE: 'CPF do Cliente',
  VALOR_RESTRICAO: 'Valor da Restrição',
  CODIGO_CONTRATO: 'Código do Contrato',
  DATA_RESTRICAO: 'Data da Restrição',
  ORGAO_RESTRITIVO: 'Órgão Restritivo',
  EMPRESA_RE: 'Empresa Ré',
  COMARCA: 'Comarca',
  NUMERO_CNJ: 'Número CNJ',
};

export function validatePlaceholders(
  template: string,
  processo: Record
): PlaceholderValidation {
  const placeholders = extractPlaceholders(template);
  const missing = placeholders
    .filter(p => {
      const val = processo[p.toLowerCase()] ?? processo[toCamelCase(p)];
      return !val;
    })
    .map(p => ({ placeholder: p, label: PLACEHOLDER_LABEL[p] ?? p }));
  return { valid: missing.length === 0, missing };
}
