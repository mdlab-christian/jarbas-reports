━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

function htmlToMd(html) {
  return html
    .replace(//gi, '\n')
    .replace(/<\/p>/gi, '\n\n')
    .replace(/]*>/gi, '')
    .replace(/(.*?)<\/strong>/gi, '**$1**')
    .replace(/(.*?)<\/b>/gi, '**$1**')
    .replace(/(.*?)<\/em>/gi, '*$1*')
    .replace(/]*>(.*?)<\/h1>/gi, '# $1\n')
    .replace(/]*>(.*?)<\/h2>/gi, '## $1\n')
    .replace(/]*>(.*?)<\/h3>/gi, '### $1\n')
    .replace(/]*>(.*?)<\/li>/gi, '- $1\n')
    .replace(/<[^>]+>/g, '')
    .replace(/&/g,'&').replace(/</g,'<').replace(/>/g,'>')
    .replace(/ /g,' ').replace(/é/g,'é').replace(/ç/g,'ç')
    .replace(/ã/g,'ã').replace(/õ/g,'õ').replace(/í/g,'í')
    .replace(/ó/g,'ó').replace(/ú/g,'ú').replace(/á/g,'á')
    .replace(/ê/g,'ê').replace(/ô/g,'ô').replace(/â/g,'â')
    .replace(/Ã/g,'Ã').replace(/É/g,'É').replace(/Á/g,'Á')
    .replace(/\n{3,}/g,'\n\n').trim();
}

async function getToken() {
  const r = await fetch('https://oauth2.googleapis.com/token', {
    method:'POST', headers:{'Content-Type':'application/x-www-form-urlencoded'},
    body: new URLSearchParams({client_id:process.env.GOOGLE_CLIENT_ID,client_secret:process.env.GOOGLE_CLIENT_SECRET,refresh_token:process.env.GOOGLE_REFRESH_TOKEN,grant_type:'refresh_token'})
  });
  return (await r.json()).access_token;
}

async function exportDrive(docId, token) {
  const r = await fetch(`https://www.googleapis.com/drive/v3/files/${docId}/export?mimeType=text/html`, {headers:{Authorization:`Bearer ${token}`}});
  if (r.status === 429) { await new Promise(r=>setTimeout(r,10000)); return exportDrive(docId, await getToken()); }
  if (!r.ok) { appendFileSync(ERROS, `ERRO_MODELO ${docId} HTTP${r.status}\n`); return null; }
  return r.text();
}

async function uploadBucket(path, content, type='text/html') {
  const { error } = await supabase.storage.from('olivia-modelos').upload(path, Buffer.from(content,'utf8'), {contentType:type,upsert:true});
  if (error) appendFileSync(ERROS, `ERRO_UPLOAD ${path}: ${error.message}\n`);
}

    const md = htmlToMd(htmlLimpo);
    const tagsModelo = (mtMap[versao.id]||[]).sort((a,b)=>a.ordem-b.ordem).map(mt=>({...mt,tag:tagMap[mt.tag_id]})).filter(mt=>mt.tag);
    const meta = {grupo_id:g.id,nome:g.nome,tipo_peticao_id:g.tipo_peticao_id,tipo_peticao_nome:tipoMap[g.tipo_peticao_id]?.nome||'',categoria_empresa_id:g.categoria_empresa_id,categoria_empresa_nome:catMap[g.categoria_empresa_id]?.nome||'',active_version_id:versao.id,versoes:[{id:versao.id,arquivo:'v1.html',google_doc_id:versao.google_doc_id}],tags:tagsModelo.map(mt=>({tag_id:mt.tag_id,codigo:mt.tag.codigo,nome:mt.tag.nome,obrigatoria:mt.obrigatoria,ordem:mt.ordem})),migrado_em:new Date().toISOString()};
    await uploadBucket(`modelos/${g.id}/v1.html`, htmlLimpo);
    await uploadBucket(`modelos/${g.id}/v1.md`, md, 'text/markdown');
    await uploadBucket(`modelos/${g.id}/meta.json`, JSON.stringify(meta,null,2), 'application/json');
    const vaultMd = `---\ngrupo_id: ${g.id}\ntipo: ${meta.tipo_peticao_nome}\ncategoria: ${meta.categoria_empresa_nome}\ntags_count: ${tagsModelo.length}\n---\n\n# ${g.nome}\n**Tipo:** ${meta.tipo_peticao_nome} | **Categoria:** ${meta.categoria_empresa_nome}\n\n## Tags\n${tagsModelo.map(mt=>`- [${mt.tag.codigo}] ${mt.tag.nome}${mt.obrigatoria?' (obrigatória)':''}`).join('\n')}\n`;
    writeFileSync(`${VAULT}/modelos/${g.id}.md`, vaultMd);
    writeFileSync(`${DESK}/modelos/${g.nome.replace(/[\/\\]/g,'_').slice(0,60)}.html`, htmlLimpo);
    ok++;
    if (ok%10===0) console.log(`✅ ${ok}/${grupos.length} modelos`);
    await new Promise(r=>setTimeout(r,500));
  }
  console.log(`\n✅ BLOCO B: ${ok} OK | ${skip} pulados. Ver ${ERROS} para erros.`);
}
main().catch(console.error);
