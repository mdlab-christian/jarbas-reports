## CRIAR hooks/useOliviaProcessoBusca.ts
```typescript
import { useQuery } from '@tanstack/react-query';
import { useState, useEffect } from 'react';
import { supabase } from '@/integrations/supabase/client';

function useDebounce(value: T, delay: number): T {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const t = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(t);
  }, [value, delay]);
  return debounced;
}

export function useOliviaProcessoBusca(search: string, orgId: string | null) {
  const debouncedSearch = useDebounce(search, 300);
  return useQuery({
    queryKey: ['olivia-processos-busca', debouncedSearch, orgId],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('processos')
        .select(`id, numero_cnj, status, tipo_processo, restricao_valor,
                 restricao_orgao_nome, reu_nome, cidade_distribuicao,
                 clientes(nome), empresas(nome, categoria_id),
                 advogados(nome)`)
        .eq('organizacao_id', orgId!)
        .is('deleted_at', null)
        .or(`numero_cnj.ilike.%${debouncedSearch}%,clientes.nome.ilike.%${debouncedSearch}%`)
        .order('updated_at', { ascending: false })
        .limit(20);
      if (error) throw error;
      return data ?? [];
    },
    enabled: !!orgId && debouncedSearch.length >= 2,
    staleTime: 30_000,
  });
}
```

## TAREFA 2: Criar hooks/useWhisperGravacao.ts
```typescript
export function useWhisperGravacao() {
  const [gravando, setGravando] = useState(false);
  const [transcrevendo, setTranscrevendo] = useState(false);
  const mediaRecorderRef = useRef(null);
  const chunksRef = useRef([]);

  const iniciarGravacao = useCallback(async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
      const mr = new MediaRecorder(stream, { mimeType: 'audio/webm' });
      chunksRef.current = [];
      mr.ondataavailable = (e) => { if (e.data.size > 0) chunksRef.current.push(e.data); };
      mr.start();
      mediaRecorderRef.current = mr;
      setGravando(true);
    } catch {
      toast.error('Microfone não disponível');
    }
  }, []);

  const pararGravacao = useCallback((): Promise => {
    return new Promise((resolve, reject) => {
      const mr = mediaRecorderRef.current;
      if (!mr) return reject('sem gravador');
      mr.onstop = async () => {
        setGravando(false);
        setTranscrevendo(true);
        try {
          const blob = new Blob(chunksRef.current, { type: 'audio/webm' });
          const formData = new FormData();
          formData.append('file', blob, 'audio.webm');
          formData.append('model', 'whisper-1');
          formData.append('language', 'pt');
          // Tentar via Supabase Edge Function primeiro, fallback toast
          const { data, error } = await supabase.functions.invoke('whisper-transcribe', { body: formData });
          if (error) throw error;
          resolve(data?.text ?? '');
        } catch {
          toast.error('Transcrição indisponível no momento');
          resolve('');
        } finally {
          setTranscrevendo(false);
          mr.stream.getTracks().forEach(t => t.stop());
        }
      };
      mr.stop();
    });
  }, []);

  return { gravando, transcrevendo, iniciarGravacao, pararGravacao };
}
```
