```typescript
export function useOliviaProcessoBusca(search: string, orgId: string | null) {
  const debouncedSearch = useDebounce(search, 300); // implementar useDebounce inline
  return useQuery({
    queryKey: ['olivia-processos-busca', debouncedSearch, orgId],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('processos')
        .select(`
          id, numero_cnj, status, tipo_processo, advogado_id,
          restricao_valor, restricao_orgao_nome, reu_nome,
          cidade_distribuicao, grau, objeto,
          clientes(nome, cpf),
          empresas(nome, categoria_id),
          advogados(nome)
        `)
        .eq('organizacao_id', orgId!)
        .is('deleted_at', null)
        .or(`numero_cnj.ilike.%${debouncedSearch}%,clientes.nome.ilike.%${debouncedSearch}%`)
        .order('updated_at', { ascending: false })
        .limit(20);
      if (error) throw error;
      return data ?? [];
    },
    enabled: !!orgId && debouncedSearch.length >= 2,
    staleTime: 30_000,
  });
}
```

## TAREFA 2: Criar `useWhisperGravacao.ts`
Caminho: src/pages/olivia/hooks/useWhisperGravacao.ts

```typescript
// Hook para gravação de voz → transcrição via OpenAI Whisper
// Usa MediaRecorder API nativa do browser
export function useWhisperGravacao() {
  const [gravando, setGravando] = useState(false);
  const [transcrevendo, setTranscrevendo] = useState(false);
  const [erro, setErro] = useState(null);
  const mediaRecorderRef = useRef(null);
  const chunksRef = useRef([]);

  const iniciarGravacao = useCallback(async () => {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    const mr = new MediaRecorder(stream, { mimeType: 'audio/webm' });
    chunksRef.current = [];
    mr.ondataavailable = (e) => { if (e.data.size > 0) chunksRef.current.push(e.data); };
    mr.start();
    mediaRecorderRef.current = mr;
    setGravando(true);
  }, []);

  const pararGravacao = useCallback((): Promise => {
    return new Promise((resolve, reject) => {
      const mr = mediaRecorderRef.current;
      if (!mr) return reject('sem gravador');
      mr.onstop = async () => {
        setGravando(false);
        setTranscrevendo(true);
        try {
          const blob = new Blob(chunksRef.current, { type: 'audio/webm' });
          const formData = new FormData();
          formData.append('file', blob, 'audio.webm');
          formData.append('model', 'whisper-1');
          formData.append('language', 'pt');
          // Usa Supabase Edge Function proxy (não expõe OPENAI_API_KEY no frontend)
          const { data, error } = await supabase.functions.invoke('whisper-transcribe', {
            body: formData,
          });
          if (error) throw error;
          resolve(data.text ?? '');
        } catch (err) {
          setErro(err instanceof Error ? err.message : 'Erro na transcrição');
          reject(err);
        } finally {
          setTranscrevendo(false);
          mr.stream.getTracks().forEach(t => t.stop());
        }
      };
      mr.stop();
    });
  }, []);

  return { gravando, transcrevendo, erro, iniciarGravacao, pararGravacao };
}
```
