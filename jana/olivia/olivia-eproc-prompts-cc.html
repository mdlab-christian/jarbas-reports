```js
// olivia-eproc-auto.mjs — Skill S0.5 — Acesso automático ao EPROC para OlivIA
// Input: { historico_id, organizacao_id, processo_id, numero_cnj, tipo_peticao_id, advogado_oab }
// Output: { sucesso, documentos_baixados, motivo_falha? }
// NUNCA usar fetch() — sempre require("https")

import { createRequire } from "module";
const require = createRequire(import.meta.url);
const https = require("https");
const http = require("http");
const fs = require("fs");
const path = require("path");

// lerEnv, httpsRequest helpers (padrão das outras skills)

// Mapa de documentos prioritários por tipo de petição
// Chave: nome parcial do tipo (lowercase) | Valor: prefixos a baixar
const DOCS_POR_TIPO = {
  'réplica':       ['CONT', 'CONTR', 'PROC', 'RG', 'END', 'ANEXO'],
  'replica':       ['CONT', 'CONTR', 'PROC', 'RG', 'END', 'ANEXO'],
  'contestação':   ['INIC', 'PET', 'PROC', 'RG', 'END', 'COMP'],
  'contestacao':   ['INIC', 'PET', 'PROC', 'RG', 'END', 'COMP'],
  'apelação':      ['SENT', 'DEC', 'DESPADEC', 'ACORD', 'ATOORD'],
  'apelacao':      ['SENT', 'DEC', 'DESPADEC', 'ACORD', 'ATOORD'],
  'impugnação':    ['SENT', 'CALC', 'MEM', 'ACORD'],
  '_default':      ['SENT', 'CONT', 'INIC', 'PET', 'DEC', 'DESPADEC', 'CONTR'],
};

// Cache de sessões por OAB (evitar re-login)
const sessaoCache = new Map();

export async function executar(input) {
  const {
    historico_id, organizacao_id, processo_id,
    numero_cnj, tipo_peticao_id, advogado_oab,
    tipo_peticao_nome = '',
  } = input;

  const env = lerEnv();
  const authH = { apikey: env.SUPABASE_KEY, Authorization: `Bearer ${env.SUPABASE_KEY}` };

  // 1. Validar inputs obrigatórios
  if (!numero_cnj) return { sucesso: false, motivo_falha: 'numero_cnj ausente', documentos_baixados: 0 };
  if (!advogado_oab) return { sucesso: false, motivo_falha: 'advogado_oab ausente', documentos_baixados: 0 };

  // 2. Atualizar progresso
  if (historico_id) {
    await atualizarProgresso(env, historico_id, 3, 'Acessando o EPROC...');
  }

  // 3. Buscar credenciais do advogado no banco
  let conta;
  try {
    const r = await httpsRequest("GET",
      `${env.SUPABASE_URL}/rest/v1/jarbas_accounts?oab=eq.${encodeURIComponent(advogado_oab)}&ativo=eq.true&select=oab,login_eproc,senha_hash,totp_secret,usa_2fa,eproc_url,sso_url&limit=1`,
      null, authH);
    conta = r.body?.[0];
  } catch(e) { console.warn('[eproc-auto] Erro ao buscar conta:', e.message); }

  if (!conta) return { sucesso: false, motivo_falha: `conta OAB ${advogado_oab} não encontrada`, documentos_baixados: 0 };

  // 4. Login com cache de sessão
  let loginResult;
  const cachedSession = sessaoCache.get(advogado_oab);
  if (cachedSession && cachedSession.page && !cachedSession.expirado) {
    console.log(`[eproc-auto] Reutilizando sessão para ${advogado_oab}`);
    loginResult = cachedSession;
  } else {
    try {
      const { loginEproc } = await import('/Users/jarbas/jarbas/src/skills/skill_login_eproc.mjs');
      loginResult = await loginEproc({
        login_eproc: conta.login_eproc,
        senha_hash: conta.senha_hash,
        tribunal: conta.tribunal || 'TJRS',
        eproc_url: conta.eproc_url || 'https://eproc1g.tjrs.jus.br',
        totp_secret: conta.totp_secret,
        usa_2fa: conta.usa_2fa,
        id: null,
      });
      if (loginResult?.page) {
        sessaoCache.set(advogado_oab, { ...loginResult, expirado: false });
        setTimeout(() => { const s = sessaoCache.get(advogado_oab); if (s) s.expirado = true; }, 20 * 60 * 1000); // 20min
      }
    } catch(e) {
      console.error('[eproc-auto] Erro login:', e.message);
      return { sucesso: false, motivo_falha: `login falhou: ${e.message}`, documentos_baixados: 0 };
    }
  }

  if (!loginResult?.page) return { sucesso: false, motivo_falha: 'login EPROC falhou', documentos_baixados: 0 };

  const page = loginResult.page;

  // 5. Navegar ao processo pelo campo de pesquisa rápida
  if (historico_id) await atualizarProgresso(env, historico_id, 6, `Navegando ao processo ${numero_cnj}...`);

  try {
    await page.fill('#txtNumProcessoPesquisaRapida', numero_cnj);
    await page.keyboard.press('Enter');
    await page.waitForLoadState('domcontentloaded', { timeout: 15000 });
    await page.waitForTimeout(2000);
  } catch(e) {
    return { sucesso: false, motivo_falha: `navegação falhou: ${e.message}`, documentos_baixados: 0 };
  }

  const url = page.url();
  if (!url.includes('processo_selecionar')) {
    return { sucesso: false, motivo_falha: 'processo não encontrado no EPROC', documentos_baixados: 0 };
  }

  // 6. Extrair links de documentos da página
  const docLinks = await page.evaluate(() => {
    const links = [];
    document.querySelectorAll('a[href]').forEach(a => {
      const href = a.href || '';
      const text = (a.textContent?.trim() || '').replace(/\s+/g, ' ');
      if (href.includes('acao=acessar_documento')) {
        links.push({ label: text, href });
      }
    });
    return links;
  });

  // 7. Filtrar por tipo de petição
  const tipoKey = tipo_peticao_nome.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '');
  const prefixos = DOCS_POR_TIPO[tipoKey] || DOCS_POR_TIPO['_default'];
  const docsAlvo = docLinks
    .filter(d => prefixos.some(p => d.label.toUpperCase().startsWith(p)))
    .slice(0, 8); // máximo 8 docs

  if (docsAlvo.length === 0) {
    // Fallback: pegar os primeiros 5 docs P1
    const p1 = ['SENT', 'CONT', 'DEC', 'DESPADEC', 'INIC'];
    docsAlvo.push(...docLinks.filter(d => p1.some(p => d.label.toUpperCase().startsWith(p))).slice(0, 5));
  }

  console.log(`[eproc-auto] ${docLinks.length} docs totais | ${docsAlvo.length} alvo para tipo "${tipoKey}"`);
  if (historico_id) await atualizarProgresso(env, historico_id, 10, `Baixando ${docsAlvo.length} documentos...`);

  // 8. Baixar e salvar cada documento
  const { downloadDocumentosEproc } = await import('/Users/jarbas/jarbas/src/skills/skill_download_documentos_eproc.mjs');

  let documentosBaixados = 0;
  for (const doc of docsAlvo) {
    try {
      // Navegar para o documento e baixar
      await page.goto(doc.href, { timeout: 15000, waitUntil: 'domcontentloaded' });
      await page.waitForTimeout(1500);

      // Tentar baixar como PDF
      const pdfBuffer = await page.pdf({ format: 'A4', printBackground: true }).catch(() => null);
      if (!pdfBuffer || pdfBuffer.length < 1000) continue;

      const nomeArquivo = sanitizarNome(doc.label);
      const caminhoTmp = `/tmp/olivia-eproc-${Date.now()}-${nomeArquivo}.pdf`;
      fs.writeFileSync(caminhoTmp, pdfBuffer);

      // Upload para olivia-docs
      const uploaded = await uploadParaOliviaDocs(env, caminhoTmp, nomeArquivo, processo_id, organizacao_id);
      if (uploaded?.arquivo_url) {
        // Inserir em processos_documentos
        await httpsRequest("POST", `${env.SUPABASE_URL}/rest/v1/processos_documentos`,
          {
            processo_id, organizacao_id,
            nome: doc.label,
            arquivo_url: uploaded.arquivo_url,
            categoria: 'eproc_auto',
            tipo: 'pdf',
          },
          { ...authH, Prefer: 'return=minimal' }
        );
        // Disparar extração de texto (fire and forget)
        fetch(`http://localhost:19100/olivia/extrair-texto`, {
          method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer olivia-gateway-token-2026' },
          body: JSON.stringify({ arquivo_url: uploaded.arquivo_url }),
        }).catch(() => {});

        documentosBaixados++;
        fs.unlinkSync(caminhoTmp);
      }
    } catch(e) {
      console.warn(`[eproc-auto] Falha ao baixar ${doc.label}:`, e.message);
    }
  }

  if (historico_id) await atualizarProgresso(env, historico_id, 15, `${documentosBaixados} documentos baixados do EPROC`);
  console.log(`[eproc-auto] CONCLUÍDO: ${documentosBaixados}/${docsAlvo.length} docs baixados`);

  return { sucesso: true, documentos_baixados: documentosBaixados, docs_totais_processo: docLinks.length };
}

// Funções auxiliares: lerEnv, httpsRequest, atualizarProgresso, sanitizarNome, uploadParaOliviaDocs
// (usar o padrão das outras skills da OlivIA)
```
