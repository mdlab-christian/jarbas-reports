-- 1. Inventário tabelas olivia_*

       (SELECT count(*) FROM information_schema.columns WHERE table_name = t.table_name) as col_count
FROM information_schema.tables t
WHERE table_schema = 'public' AND table_name LIKE 'olivia_%'
ORDER BY table_name;

-- 2. Contagem de dados críticos
SELECT 'olivia_modelos_peticao' as t, count(*) FROM olivia_modelos_peticao WHERE ativo = true
UNION ALL SELECT 'olivia_modelos_grupo', count(*) FROM olivia_modelos_grupo WHERE deleted_at IS NULL
UNION ALL SELECT 'olivia_tags', count(*) FROM olivia_tags WHERE ativo = true
UNION ALL SELECT 'olivia_blocos_texto', count(*) FROM olivia_blocos_texto WHERE ativo = true
UNION ALL SELECT 'olivia_juris', count(*) FROM olivia_jurisprudencias WHERE ativo = true
UNION ALL SELECT 'olivia_modelo_tags', count(*) FROM olivia_modelo_tags
UNION ALL SELECT 'olivia_modelo_juris', count(*) FROM olivia_modelo_juris
UNION ALL SELECT 'olivia_flow', count(*) FROM olivia_flow
UNION ALL SELECT 'olivia_historico', count(*) FROM olivia_historico;

-- 3. published_at status
SELECT 
  count(*) FILTER (WHERE published_at IS NOT NULL) as com_published,
  count(*) FILTER (WHERE published_at IS NULL) as sem_published,
  count(*) as total
FROM olivia_modelos_peticao WHERE ativo = true;

-- 4. FK problema histórico
SELECT column_name, data_type, is_nullable
FROM information_schema.columns
WHERE table_name = 'olivia_historico'
ORDER BY ordinal_position;

-- 5. Etapas do projeto
SELECT ordem, titulo, status, fase
FROM projetos_etapas
WHERE projeto_id = '48d9bfaa-206f-457c-95f6-36ed25e4162e'
