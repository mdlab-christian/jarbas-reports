Changelog v4.0 (JANA ?? — 2026-03-16) ?? * Retorno à abordagem do GAS antigo: copia modelo + substitui blocos de docs * Preserva toda a formatação dos docs-fonte (parágrafos, tabelas, listas, imgs) * Mantém melhorias do v3.2: retry, idempotência, tags de dados, resposta rica * FIX-C2: moveTo() em vez de addFile/removeFile (API deprecada) * FIX-NEW: substituição de {{tags}} dados APÓS substituição dos blocos * FIX-NEW: pasta destino por ID direto (não path string) * FIX-NEW: suporte a substituição de tags em header/footer do doc * FIX-NEW: idempotência por numero_processo + modelo_id (não título) * ??????????????????????????????????????????????????????????????????????????????? */// ???????????????????????????????????????????????????????????????????????????????// SECTION 1: ENTRY POINTS// ???????????????????????????????????????????????????????????????????????????????/** * POST handler — ponto de entrada principal. */function doPost(e) {  var startTime = new Date().getTime();  try {    var payload = JSON.parse(e.postData.contents);    var debug   = payload.configuracoes && payload.configuracoes.debug === true;    if (debug) Logger.log('[DEBUG] Payload recebido: ' + JSON.stringify(payload).substring(0, 2000));    // Validar payload    var validation = validatePayload(payload);    if (!validation.valid) {      return jsonResponse({ success: false, error_code: 'VALIDATION_ERROR', error_message: validation.error });    }    var processo      = payload.processo      || {};    var modelo        = payload.modelo;    var blocos        = payload.blocos        || [];    var jurisprudencias = payload.jurisprudencias || [];    var tags          = payload.tags_substituicao || {};    var config        = payload.configuracoes || {};    var nomeArquivo   = config.nome_arquivo || ('Petição - ' + (processo.numero || new Date().toISOString()));    var pastaId       = config.pasta_destino_id || null;    var fallbackRoot  = config.fallback_to_root === true;    // Idempotência: chave = numero_processo + modelo_id (janela 60s)    var idempotencyKey = generateIdempotencyKey(      (processo.numero || nomeArquivo),      modelo.google_doc_id    );    var cachedResult = checkDuplicate(idempotencyKey);    if (cachedResult) {      if (debug) Logger.log('[DEBUG] Duplicata detectada — retornando cache');      cachedResult.from_cache = true;      return jsonResponse(cachedResult);    }    // 1. Copiar o modelo    if (debug) Logger.log('[DEBUG] Copiando modelo: ' + modelo.google_doc_id);    var copiaResult = copiarModelo(modelo.google_doc_id, nomeArquivo, pastaId, fallbackRoot);    if (!copiaResult.success) {      return jsonResponse({ success: false, error_code: copiaResult.error_code, error_message: copiaResult.error_message });    }    var doc   = copiaResult.doc;    var copiaId = doc.getId();    var body  = doc.getBody();    // 2. Substituir blocos (de baixo para cima — preserva índices)    var blocosOrdenados = blocos.slice().sort(function(a, b) { return b.ordem - a.ordem; });    var blocosOk = 0, blocosFail = [];    for (var i = 0; i < blocosOrdenados.length; i++) {      var bloco = blocosOrdenados[i];      if (debug) Logger.log('[DEBUG] Substituindo bloco: ' + bloco.placeholder);      var ok = substituirConteudo(body, bloco.placeholder, bloco.google_doc_id, debug);      if (ok) { blocosOk++; } else { blocosFail.push(bloco.placeholder); }    }    // 3. Substituir jurisprudências    var jurisOk = 0, jurisFail = [];    for (var j = 0; j < jurisprudencias.length; j++) {      var juris = jurisprudencias[j];      if (debug) Logger.log('[DEBUG] Substituindo jurisprudência: ' + juris.placeholder);      var jok = substituirConteudo(body, juris.placeholder, juris.google_doc_id, debug);      if (jok) { jurisOk++; } else { jurisFail.push(juris.placeholder); }    }    // 4. Substituir {{tags}} de dados (body + header + footer)    var tagStats = { substituidas: 0, nao_encontradas: [], residuais_marcadas: 0 };    replaceTagsInSection(body, tags, tagStats);    var header = doc.getHeader();    if (header) replaceTagsInSection(header, tags, tagStats);    var footer = doc.getFooter();    if (footer) replaceTagsInSection(footer, tags, tagStats);    // 5. Limpeza de tags residuais não substituídas    cleanResidualTags(doc, tagStats);    // 6. Salvar e fechar    doc.saveAndClose();    var executionTime = new Date().getTime() - startTime;    var result = {      success: true,      google_doc_id:    copiaId,      google_doc_url:   'https://docs.google.com/document/d/' + copiaId + '/edit',      nome_arquivo:     nomeArquivo,      estatisticas: {        blocos_substituidos:         blocosOk,        blocos_nao_encontrados:      blocosFail,        jurisprudencias_substituidas: jurisOk,        jurisprudencias_nao_encontradas: jurisFail,        tags_substituidas:           tagStats.substituidas,        tags_nao_encontradas:        tagStats.nao_encontradas,        tags_residuais_marcadas:     tagStats.residuais_marcadas,        tempo_processamento_ms:      executionTime      }    };    saveDuplicateKey(idempotencyKey, result);    if (debug) Logger.log('[DEBUG] Concluído: ' + copiaId + ' em ' + executionTime + 'ms');    return jsonResponse(result);  } catch (error) {    Logger.log('[ERROR] ' + error.message + '\n' + (error.stack || ''));    return jsonResponse({      success: false,      error_code: 'EXECUTION_ERROR',      error_message: error.message,      error_stack: error.stack ? error.stack.substring(0, 500) : null    });  }}/** * GET handler — health check. */function doGet(e) {  return jsonResponse({    success: true,    service: 'MdFlow Google Doc Creator',    version: '4.0',    timestamp: new Date().toISOString(),    message: 'OK — POST para criar documento'  });}// ???????????????????????????????????????????????????????????????????????????????// SECTION 2: VALIDATION// ???????????????????????????????????????????????????????????????????????????????function validatePayload(payload) {  if (!payload)    return { valid: false, error: 'Payload vazio ou inválido' };  if (!payload.modelo || !payload.modelo.google_doc_id)    return { valid: false, error: 'modelo.google_doc_id é obrigatório' };  if (payload.blocos && !Array.isArray(payload.blocos))    return { valid: false, error: 'blocos deve ser array' };  if (payload.jurisprudencias && !Array.isArray(payload.jurisprudencias))    return { valid: false, error: 'jurisprudencias deve ser array' };  var todos = (payload.blocos || []).concat(payload.jurisprudencias || []);  for (var i = 0; i < todos.length; i++) {    var item = todos[i];    if (!item.placeholder)      return { valid: false, error: 'Cada bloco/jurisprudência deve ter campo "placeholder"' };    if (!item.google_doc_id)      return { valid: false, error: 'Cada bloco/jurisprudência deve ter campo "google_doc_id"' };  }  return { valid: true, error: null };}// ???????????????????????????????????????????????????????????????????????????????// SECTION 3: CÓPIA DO MODELO// ???????????????????????????????????????????????????????????????????????????????function copiarModelo(modeloId, nomeArquivo, pastaId, fallbackRoot) {  return withRetry(function() {    var modeloFile;    try {      modeloFile = DriveApp.getFileById(modeloId);      modeloFile.getName(); // testa permissão    } catch (e) {      return {        success: false, doc: null,        error_code: 'MODEL_ACCESS_DENIED',        error_message: 'Sem acesso ao modelo ' + modeloId + ': ' + e.message      };    }    // Criar cópia    var copiaFile = modeloFile.makeCopy(nomeArquivo);    var doc = DocumentApp.openById(copiaFile.getId());    // Mover para pasta destino    if (pastaId) {      try {        var pasta = DriveApp.getFolderById(pastaId);        pasta.getName(); // testa permissão        copiaFile.moveTo(pasta); // FIX-C2      } catch (e) {        if (!fallbackRoot) {          // Apagar cópia já criada antes de retornar erro          try { copiaFile.setTrashed(true); } catch(ex) {}          return {            success: false, doc: null,            error_code: 'FOLDER_ACCESS_DENIED',            error_message: 'Sem acesso à pasta ' + pastaId + ': ' + e.message          };        }        Logger.log('[WARN] Pasta inacessível, usando raiz: ' + e.message);      }    }    return { success: true, doc: doc, error_code: null, error_message: null };  }, 2, 1500);}// ???????????????????????????????????????????????????????????????????????????????// SECTION 4: SUBSTITUIÇÃO DE BLOCO (PRESERVA FORMATAÇÃO)// ???????????????????????????????????????????????????????????????????????????????/** * Localiza o placeholder no bodyDestino, abre o doc-fonte e copia * o conteúdo preservando formatação (parágrafos, tabelas, listas, imagens). */function substituirConteudo(bodyDestino, placeholder, googleDocId, debug) {  try {    var searchResult = bodyDestino.findText(escapeRegex(placeholder));    if (!searchResult) {      Logger.log('[WARN] Placeholder não encontrado: ' + placeholder);      return false;    }    // Abrir doc-fonte    var fonteDoc  = DocumentApp.openById(googleDocId);    var fonteBody = fonteDoc.getBody();    // Determinar posição de inserção    var element      = searchResult.getElement();    var startOffset  = searchResult.getStartOffset();    var endOffset    = searchResult.getEndOffsetInclusive();    var paragrafoPai = element.getParent();    var indice       = bodyDestino.getChildIndex(paragrafoPai);    // Remover o placeholder do parágrafo    element.asText().deleteText(startOffset, endOffset);    // Copiar conteúdo formatado    copiarConteudoFormatado(bodyDestino, fonteBody, indice, debug);    // Se o parágrafo que tinha o placeholder ficou vazio, remover    try {      var paraAtual = bodyDestino.getChild(indice);      if (paraAtual && paraAtual.getType() === DocumentApp.ElementType.PARAGRAPH) {        if (paraAtual.asText && paraAtual.asText().getText().trim() === '') {          bodyDestino.removeChild(paraAtual);        }      }    } catch (e) { /* ignorar — parágrafo pode já ter sido deslocado */ }    return true;  } catch (erro) {    Logger.log('[ERROR] substituirConteudo(' + placeholder + '): ' + erro.message);    return false;  }}/** * Copia todos os elementos do bodyOrigem para bodyDestino a partir de posicaoInsercao+1, * preservando toda a formatação original (copy() nativo do Apps Script). */function copiarConteudoFormatado(bodyDestino, bodyOrigem, posicaoInsercao, debug) {  var numElementos    = bodyOrigem.getNumChildren();  var elementosCopiados = 0;  for (var i = 0; i < numElementos; i++) {    var elemento = bodyOrigem.getChild(i);    var tipo     = elemento.getType();    var posAjustada = posicaoInsercao + 1 + i;    try {      switch (tipo) {        case DocumentApp.ElementType.PARAGRAPH:          var paragrafo = elemento.asParagraph().copy();          bodyDestino.insertParagraph(posAjustada, paragrafo);          elementosCopiados++;          break;        case DocumentApp.ElementType.TABLE:          var tabela = elemento.asTable().copy();          bodyDestino.insertTable(posAjustada, tabela);          elementosCopiados++;          break;        case DocumentApp.ElementType.LIST_ITEM:          var listItem = elemento.asListItem().copy();          bodyDestino.insertListItem(posAjustada, listItem);          elementosCopiados++;          break;        default:          Logger.log('[INFO] Tipo de elemento ignorado: ' + tipo);          break;      }    } catch (e) {      Logger.log('[WARN] Erro ao copiar elemento ' + i + ' (' + tipo + '): ' + e.message);    }  }  if (debug) Logger.log('[DEBUG]   ? ' + elementosCopiados + ' elementos copiados');  return elementosCopiados;}// ???????????????????????????????????????????????????????????????????????????????// SECTION 5: SUBSTITUIÇÃO DE TAGS {{dados}}// ???????????????????????????????????????????????????????????????????????????????/** * Substitui {{tags}} diretamente no documento, preservando formatação do texto. */function replaceTagsInSection(section, tags, tagStats) {  for (var tagName in tags) {    if (!tags.hasOwnProperty(tagName)) continue;    var valor = tags[tagName];    if (valor == null) continue;    var valorStr = String(valor);    if (valorStr.trim() === '') continue;    // Padrão com underscores e com pontos (compatibilidade)    var patterns = ['\\{\\{' + escapeRegex(tagName) + '\\}\\}'];    if (tagName.indexOf('_') !== -1) {      patterns.push('\\{\\{' + escapeRegex(tagName.replace(/_/g, '.')) + '\\}\\}');    }    for (var p = 0; p < patterns.length; p++) {      var searchResult = section.findText(patterns[p]);      var safety = 0;      while (searchResult && safety < 200) {        var elem  = searchResult.getElement().asText();        var start = searchResult.getStartOffset();        var end   = searchResult.getEndOffsetInclusive();        var attrs = captureTextAttributes(elem, start);        elem.deleteText(start, end);        elem.insertText(start, valorStr);        var newEnd = start + valorStr.length - 1;        if (newEnd >= start) applyTextAttributes(elem, start, newEnd, attrs);        tagStats.substituidas++;        searchResult = section.findText(patterns[p], searchResult);        safety++;      }    }  }}function captureTextAttributes(elem, offset) {  return {    bold:            elem.isBold(offset),    italic:          elem.isItalic(offset),    underline:       elem.isUnderline(offset),    fontSize:        elem.getFontSize(offset),    fontFamily:      elem.getFontFamily(offset),    foregroundColor: elem.getForegroundColor(offset)  };}function applyTextAttributes(elem, start, end, attrs) {  if (attrs.bold            !== null) elem.setBold(start, end, attrs.bold);  if (attrs.italic          !== null) elem.setItalic(start, end, attrs.italic);  if (attrs.underline       !== null) elem.setUnderline(start, end, attrs.underline);  if (attrs.fontSize        !== null) elem.setFontSize(start, end, attrs.fontSize);  if (attrs.fontFamily      !== null) elem.setFontFamily(start, end, attrs.fontFamily);  if (attrs.foregroundColor !== null) elem.setForegroundColor(start, end, attrs.foregroundColor);}// ???????????????????????????????????????????????????????????????????????????????// SECTION 6: LIMPEZA DE TAGS RESIDUAIS// ???????????????????????????????????????????????????????????????????????????????function cleanResidualTags(doc, tagStats) {  var sections = [doc.getBody()];  var header = doc.getHeader();  if (header) sections.push(header);  var footer = doc.getFooter();  if (footer) sections.push(footer);  for (var s = 0; s < sections.length; s++) {    cleanResidualTagsInSection(sections[s], tagStats);  }}function cleanResidualTagsInSection(section, tagStats) {  var pattern = '\\{\\{[^}]+\\}\\}';  var searchResult = section.findText(pattern);  var safety = 0;  while (searchResult && safety < 100) {    var elem  = searchResult.getElement().asText();    var start = searchResult.getStartOffset();    var end   = searchResult.getEndOffsetInclusive();    var originalText = elem.getText().substring(start, end + 1);    var tagName = originalText.replace(/\{\{|\}\}/g, '').trim();    if (tagStats.nao_encontradas.indexOf(tagName) === -1) {      tagStats.nao_encontradas.push(tagName);    }    var replacement = '[REVISAR: ' + tagName + ']';    elem.deleteText(start, end);    elem.insertText(start, replacement);    var newEnd = start + replacement.length - 1;    elem.setForegroundColor(start, newEnd, '#FF0000');    elem.setBold(start, newEnd, true);    elem.setFontSize(start, newEnd, 10);    tagStats.residuais_marcadas++;    searchResult = section.findText(pattern, searchResult);    safety++;  }}// ???????????????????????????????????????????????????????????????????????????????// SECTION 7: UTILITIES// ???????????????????????????????????????????????????????????????????????????????function jsonResponse(data) {  return ContentService    .createTextOutput(JSON.stringify(data))    .setMimeType(ContentService.MimeType.JSON);}function escapeRegex(str) {  return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');}function withRetry(fn, maxRetries, delayMs) {  maxRetries = maxRetries || 2;  delayMs    = delayMs    || 1500;  var lastError;  for (var attempt = 0; attempt <= maxRetries; attempt++) {    try { return fn(); } catch (e) {      lastError = e;      Logger.log('[RETRY] Tentativa ' + (attempt + 1) + '/' + (maxRetries + 1) + ' falhou: ' + e.message);      if (attempt < maxRetries) Utilities.sleep(delayMs * (attempt + 1));    }  }  throw lastError;}function generateIdempotencyKey(numeroProcesso, modeloId) {  var windowKey = Math.floor(new Date().getTime() / 60000);  var digest = Utilities.computeDigest(    Utilities.DigestAlgorithm.MD5,    numeroProcesso + '|' + modeloId + '|' + windowKey  );  var hex = '';  for (var i = 0; i < digest.length && hex.length < 16; i++) {    var byte = digest[i] & 0xFF;    hex += (byte < 16 ? '0' : '') + byte.toString(16);  }  return 'idem_' + hex.substring(0, 16);}function checkDuplicate(key) {  try {    var props  = PropertiesService.getScriptProperties();    var cached = props.getProperty(key);    if (cached) {      var parsed = JSON.parse(cached);      if (parsed._ts && (new Date().getTime() - parsed._ts) > 120000) {        props.deleteProperty(key);        return null;      }      return parsed;    }  } catch (e) { Logger.log('[WARN] checkDuplicate falhou: ' + e.message); }  return null;}function saveDuplicateKey(key, result) {  try {    var props = PropertiesService.getScriptProperties();    props.setProperty(key, JSON.stringify({      success:        result.success,      google_doc_id:  result.google_doc_id,      google_doc_url: result.google_doc_url,      nome_arquivo:   result.nome_arquivo,      estatisticas:   result.estatisticas,      _ts:            new Date().getTime()    }));  } catch (e) { Logger.log('[WARN] saveDuplicateKey falhou: ' + e.message); }}// ???????????????????????????????????????????????????????????????????????????????// SECTION 8: FUNÇÃO DE TESTE LOCAL// (Remover ou comentar antes de deploy em produção)// ???????????????????????????????????????????????????????????????????????????????function testeLocal() {  var dadosTeste = {    processo: {      numero: 'TESTE-001/2026'    },    modelo: {      google_doc_id: 'ID_DO_MODELO_DE_PETICAO'    },    blocos: [      { ordem: 1, placeholder: '{{enderecamento}}',  google_doc_id: 'ID_BLOCO_ENDERECAMENTO' },      { ordem: 2, placeholder: '{{qualificacao}}',   google_doc_id: 'ID_BLOCO_QUALIFICACAO' },      { ordem: 3, placeholder: '{{merito}}',         google_doc_id: 'ID_BLOCO_MERITO' }    ],    jurisprudencias: [      { placeholder: '{{juris_consumidor_1}}', google_doc_id: 'ID_JURIS_CONSUMIDOR' }    ],    tags_substituicao: {      cliente_nome:    'João Silva Santos',      cliente_cpf:     '123.456.789-00',      reu_nome:        'Empresa Teste S.A.',      comarca:         'Porto Alegre/RS',      valor_causa:     'R$ 15.000,00',      data_hoje:       '16 de março de 2026',      advogado_nome:   'Dr. Carlos Paz',      advogado_oab:    'OAB/RS 12345'    },    configuracoes: {      nome_arquivo:       'TESTE — Petição Inicial — João Silva',      pasta_destino_id:   'ID_DA_PASTA_DESTINO',      fallback_to_root:   true,      debug:              true    }  };  var fakeE = { postData: { contents: JSON.stringify(dadosTeste) } };  var resultado = JSON.parse(doPost(fakeE).getContent());  Logger.log('=== RESULTADO TESTE ===');  Logger.log(JSON.stringify(resultado, null, 2));}
