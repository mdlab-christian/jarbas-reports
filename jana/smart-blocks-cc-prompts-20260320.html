## Tarefa 2: Adicionar fetchGoogleDocHtml() e interpolar blocos

Adicionar antes da função criarDoc():

```javascript
/**
 * Busca conteúdo HTML de um Google Doc via Drive API
 * @param {string} googleDocId - ID do Google Doc
 * @param {object} auth - OAuth2Client autenticado
 * @returns {Promise} HTML do documento
 */
async function fetchGoogleDocHtml(googleDocId, auth) {
  const { google } = await import('googleapis');
  const drive = google.drive({ version: 'v3', auth });
  const res = await drive.files.export({
    fileId: googleDocId,
    mimeType: 'text/html',
  });
  // Limpar HTML do Google (remover head, body wrappers)
  let html = typeof res.data === 'string' ? res.data : '';
  // Extrair apenas o conteúdo do body
  const bodyMatch = html.match(/]*>([\s\S]*?)<\/body>/i);
  return bodyMatch ? bodyMatch[1].trim() : html;
}
```

    const placeholder = bloco.placeholder || `{{BLOCO_${bloco.ordem}}}`;
    
    if (bloco.google_doc_id) {
      // Buscar conteúdo real do Google Drive
      try {
        const blocoHtml = await fetchGoogleDocHtml(bloco.google_doc_id, auth);
        if (htmlFinal.includes(placeholder)) {
          htmlFinal = htmlFinal.replace(placeholder, blocoHtml);
        } else {
          // Fallback: append ao final se placeholder não encontrado
          htmlFinal += '\n\n' + blocoHtml;
        }
        console.log(`[olivia-doc-creator] Bloco ${bloco.google_doc_id} interpolado OK`);
      } catch (err) {
        console.error(`[olivia-doc-creator] Bloco ${bloco.google_doc_id} falhou:`, err.message);
        // Continuar sem o bloco (não bloquear a geração inteira)
      }
    } else if (bloco.conteudo) {
      // Bloco sem Google Doc — usar conteudo direto como parágrafo
      const blocoHtml = `
${bloco.conteudo}
`;
      if (htmlFinal.includes(placeholder)) {
        htmlFinal = htmlFinal.replace(placeholder, blocoHtml);
      } else {
        htmlFinal += '\n\n' + blocoHtml;
      }
    }
    // Se bloco sem google_doc_id e sem conteudo: ignorar silenciosamente
  }
}
