async function executarCiclo(account, horario) {
  // Etapa 1: Login EPROC (Playwright + Keycloak SSO)
  const loginResult = await loginEproc(account);

  // Etapa 2: Gerar planilha XLS
  const planilha = await skillGerarPlanilha(page, { oab, tribunal, sessionHash });

  // Etapa 3: Processar planilha + filtros inteligentes
  const processamento = await skillProcessarPlanilha({ filePath, desdeTimestamp, oab });

  // O2+O9+O10: Batch lookups (per-process timestamps + status)
  const [ultimaAnaliseMap, statusMap] = await Promise.all([
    batchLookupUltimaAnalise(allCnjs),
    batchLookupStatus(allCnjs),
  ]);

  // Filtrar: skip arquivados (O9), per-process timestamp (O2),
  // transitado_julgado so financeiros, pre-classificar (O3)

  // Etapas 4-5: Acessar processos + ler documentos
  for (const proc of precisamEproc) {
    const acesso = await skillAcessarProcesso(page, {
      cnj, sessionHash, desdeTimestamp, tribunal,
      maxEventos: 3,      // <-- PROPOSTO: mudar para 50
      hrefsLidos,          // O8: dedup entre processos
    });
    for (const evento of acesso.eventos) {
      for (const doc of evento.documentos) {
        const leitura = await skillLerDocumento(page, {
          docHref, cnj, eventoTexto, isDocCritico   // O7: Vision so para criticos
        });
      }
    }
  }

  // Etapa 6: Analise IA (Haiku triagem → Sonnet decisoes)
  const analise = await skillAnalisarIA(movimentacoesParaIA, { cicloId, oab });

  // O12: Notificacao imediata E/F
  for (const r of analise.resultados) {
    if (['E','F'].includes(r.classe)) await notificarImediato({...r, tribunal, oab}, cicloId);
  }

  // Persistir em intimacoes + atualizar ultima_analise
  for (const r of analise.resultados) await persistirIntimacao(r, ...);

  // Notificar resumo do ciclo via Telegram
  await notificarResumoCiclo(cicloData, analise.resultados);
}
Ver executarTodosCiclos() — Orquestracao de todas as OABs 
export async function executarTodosCiclos(horario) {
  const accounts = await loadAccounts();

  if (!accounts.length) {
    console.error('NENHUMA CONTA EPROC ATIVA');
    await notificarZeroContas(horario);  // P0-4: alerta imediato
    return;
  }

  // Ordenar: TJRS → TJSC → TJSP
  const ordenados = [...accounts].sort((a, b) => {
    const ordem = { TJRS: 1, TJSC: 2, TJSP: 3 };
    return (ordem[a.tribunal] || 9) - (ordem[b.tribunal] || 9);
  });

  for (const account of ordenados) {
    // Retry ate MAX_RETRY_POR_OAB (2) com backoff 30s
    for (let tentativa = 0; tentativa <= MAX_RETRY_POR_OAB; tentativa++) {
      resultado = await executarCiclo(account, horario);
      if (resultado.success) break;
    }

    // Pausa entre OABs: 5 minutos (DELAY_ENTRE_OABS_MS)
    await oabDelay();
  }

  // Notificacao final: ciclo completo
  await notificarCicloCompleto({ horario, totalOabs, totalProcessos, custoTotal, ... });
}

function mereceSonnet(mov, haikuClasse) {
  if (!['D', 'E', 'F'].includes(haikuClasse)) return false;  // So decisoes
  if (!mov.texto || mov.texto.length < 200) return false;     // Texto substancial
  if (mov.isAssinatura) return false;                         // Nao assinatura
  // Evento e decisao judicial ou financeiro
  return isDecisaoJudicial(mov.evento) || isEventoFinanceiro(mov.evento) ||
    isDecisaoJudicial(mov.texto?.slice(0, 500)) || isEventoFinanceiro(mov.texto?.slice(0, 500));
}

Problemas Encontrados

P0: Limite de 3 Eventos (maxEventos=3)

O parametro maxEventos=3 em skill_acessar_processo.mjs:17 limita a extrair apenas os 3 eventos mais recentes. Se um processo teve 10 movimentacoes desde a ultima analise, 7 serao ignoradas.

Arquivo: skill_acessar_processo.mjs linha 17 + skill_orquestrador.mjs linha 452

Fix: Mudar default para 50 (safety valve). O filtro real e desdeTimestamp.

P1: Cron de 5 Minutos Redundante

O cron */5 * * * * em start.mjs executa criarTarefasClassificacao() que busca intimacoes com ia_status='pendente'. No v2.2, todas as intimacoes sao inseridas com ia_status='concluido'. Este cron roda 288x/dia sem resultado.

Fix: Remover ou comentar.

P2: Crons Ativos sem Teste E2E

Os 3 crons de ciclo (00h, 12h, 18h BRT) estao ativos mas nenhum ciclo real com dados rodou ainda. O proximo ciclo (00h) vai rodar automaticamente sem validacao.

Fix: Pausar crons, rodar teste E2E controlado, depois reativar.

Observacoes (nao-criticos)
  • page.goto() em skill_ler_documento.mjs (linha 31) — pode quebrar sessao EPROC. Funciona para documentos porque cada doc tem hash proprio na URL, mas e um risco.
  • Tabela processos_movimentacoes (330K registros) com detectado_deposito/detectado_alvara nao utilizada. Potencial para correlacao futura.
  • Prompt Sonnet carrega 14 arquivos do vault a cada cache miss (1h TTL). Se vault for grande, pode impactar latencia.
