-- Migration: Add IA tracking columns to intimacoes
ALTER TABLE intimacoes
  ADD COLUMN IF NOT EXISTS modelo_ia text,
  ADD COLUMN IF NOT EXISTS custo_ia numeric(10,6),
  ADD COLUMN IF NOT EXISTS tokens_ia integer,
  ADD COLUMN IF NOT EXISTS analise_versao text;

-- Index for pending analysis
CREATE INDEX IF NOT EXISTS idx_intimacoes_ia_status
  ON intimacoes(ia_status)
  WHERE ia_status IS NULL OR ia_status = 'pendente';

-- Trigger: auto-analyze on INSERT
CREATE OR REPLACE FUNCTION fn_trigger_analise_intimacao()
RETURNS trigger AS $$
BEGIN
  PERFORM net.http_post(
    url := 'https://qdivfairxhdihaqqypgb.supabase.co/functions/v1/analisar-intimacao',
    headers := jsonb_build_object(
      'Authorization', 'Bearer ' || current_setting('app.settings.service_role_key'),
      'Content-Type', 'application/json'
    ),
    body := jsonb_build_object(
      'intimacao_id', NEW.id,
      'organizacao_id', NEW.organizacao_id
    )
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_analise_intimacao
  AFTER INSERT ON intimacoes
  FOR EACH ROW EXECUTE FUNCTION fn_trigger_analise_intimacao();

Ver SQL de Agregacao (jurimetria_juizes)
-- Agrega julgadores em jurimetria_juizes
INSERT INTO jurimetria_juizes (organizacao_id, julgador_nome, comarca, total_processos, total_favoraveis, taxa_favorabilidade_pct, tempo_medio_decisao_dias)
SELECT
  j.organizacao_id,
  j.nome_normalizado,
  cv.nome AS comarca,
  COUNT(d.id) AS total_processos,
  COUNT(CASE WHEN d.resultado IN ('favoravel','parcial') THEN 1 END),
  ROUND(100.0 * COUNT(CASE WHEN d.resultado IN ('favoravel','parcial') THEN 1 END) / NULLIF(COUNT(d.id), 0), 1),
  AVG(d.data_decisao - p.data_distribuicao)
FROM julgadores j
JOIN processos p ON p.julgador_id = j.id
JOIN decisoes d ON d.processo_id = p.id
LEFT JOIN comarcas_varas cv ON cv.id = j.comarca_vara_id
WHERE j.organizacao_id = '55a0c7ba...'
GROUP BY 1,2,3
ON CONFLICT (organizacao_id, julgador_nome) DO UPDATE SET
  total_processos = EXCLUDED.total_processos,
  total_favoraveis = EXCLUDED.total_favoraveis,
  taxa_favorabilidade_pct = EXCLUDED.taxa_favorabilidade_pct,
  atualizado_em = NOW();
