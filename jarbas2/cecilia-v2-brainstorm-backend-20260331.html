🔧 Como Implementar — Detalhes Técnicos

M1 — Batch lookup MdFlow (1 query antes da IA)

No orquestrador, logo após filtrar o XLS por data, fazer 1 query única com todos os CNJs:

// skill_orquestrador.mjs — nova função
async function enriquecerProcessos(processosXLS) {
  const cnjs = processosXLS.map(p => p.cnj).filter(Boolean);
  
  const { data } = await supabase
    .from('processos')
    .select(`
      numero_cnj,
      empresa_nome,
      reu_nome,
      polo_representado,
      status,
      ajg_concedida,
      fase_processual,
      ultima_movimentacao_texto
    `)
    .in('numero_cnj', cnjs)
    .eq('organizacao_id', ORG_ID);

  // Montar mapa por CNJ para lookup O(1)
  const mapa = Object.fromEntries((data || []).map(p => [p.numero_cnj, p]));
  
  // Enriquecer cada processo do XLS
  return processosXLS.map(p => ({
    ...p,
    _mdflow: mapa[p.cnj] || null
  }));
}

Custo: 1 query Supabase por OAB (independente de quantos processos). Tempo: ~50ms.

M3 — Verificar tarefa aberta recente (Regra 3)

// Batch lookup de atividades abertas
async function verificarTarefasAbertas(processoIds) {
  const cincoUteis = new Date();
  cincoUteis.setDate(cincoUteis.getDate() - 7); // ~5 dias úteis
  
  const { data } = await supabase
    .from('atividades')
    .select('processo_id, titulo, created_at, status')
    .in('processo_id', processoIds)
    .neq('status', 'concluida')
    .is('deleted_at', null)
    .gte('created_at', cincoUteis.toISOString());

  // Retorna Set de processo_ids com tarefa aberta
  return new Set((data || []).map(a => a.processo_id));
}

// No orquestrador:
const processosIds = processosEnriquecidos
  .map(p => p._mdflow?.id).filter(Boolean);
const comTarefaAberta = await verificarTarefasAbertas(processosIds);

// Filtrar ruído:
const paraProsseguir = processosEnriquecidos.filter(proc => {
  const id = proc._mdflow?.id;
  if (id && comTarefaAberta.has(id)) {
    console.log(`   ♻️  [R3] Ruído (tarefa aberta recente): ${proc.cnj}`);
    return false;
  }
  return true;
});

M5 — Roteamento automático via cecilia_equipe (ZERO ESFORÇO EXTRA)

A tabela cecilia_equipe já tem a matriz de roteamento. Só precisamos consultá-la após a classificação:

// Nova função de roteamento
async function rotearPorReu(empresaNome, tipoAtividade) {
  // Mapear empresa para tipo_reu genérico
  const tipoReu = classificarTipoReu(empresaNome);
  // Ex: "Boa Vista" → 'Boa Vista' | "Banco Santander" → 'bancos'
  
  const { data } = await supabase
    .from('cecilia_equipe')
    .select('nome, prazo_dias')
    .contains('tipo_atividade', [tipoAtividade])
    .or(`tipo_reu.cs.{"${tipoReu}"},tipo_reu.cs.{"*"}`)
    .eq('ativo', true)
    .limit(1)
    .single();

  return data || { nome: 'João Victor', prazo_dias: null };
}

// Após classificação pela IA:
if (['C', 'D', 'E'].includes(resultado.classe)) {

M6 — Criar atividade no MdFlow após classificação C/D/E/F

O João quer que as intimações apareçam como tarefas no MdFlow, não só como registros na tabela de intimações:

// Para classes C, D, E, F: criar atividade
async function criarAtividade(proc, resultado) {
  if (!['C', 'D', 'E', 'F'].includes(resultado.classe)) return;
  if (!proc._mdflow?.id) return; // sem processo no MdFlow, sem atividade
  
  const titulosPorClasse = {
    C: `📋 Documentos necessários — ${proc._mdflow.reu_nome}`,
    D: `⚖️ ${resultado.acao_sugerida || 'Providência jurídica'} — ${proc._mdflow.reu_nome}`,
    E: `🚨 ${resultado.classe === 'E' ? 'ESTRATÉGICO' : ''} — ${proc._mdflow.reu_nome}`,
    F: `💰 VALORES — ${proc._mdflow.reu_nome}`,
  };

  await supabase.from('atividades').insert({
    organizacao_id: ORG_ID,
    processo_id: proc._mdflow.id,
    titulo: titulosPorClasse[resultado.classe],
    descricao: `[Cecília] ${resultado.justificativa || resultado.acao_sugerida}\n\nEvento: ${proc.ultimoEvento}\nConfiança: ${resultado.confianca}%`,
    status: 'aberta',
    prioridade: resultado.classe === 'F' ? 'urgente' : (resultado.classe === 'E' ? 'alta' : 'normal'),
    responsavel_id: resultado.responsavel_id, // id do usuário mapeado
    data_vencimento: resultado.prazo_dias 
      ? new Date(Date.now() + resultado.prazo_dias * 86400000).toISOString().split('T')[0]
      : null,
  });
}
