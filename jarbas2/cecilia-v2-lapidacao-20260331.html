// skill_processar_planilha.mjs — adicionar filtro após parse do XLS:
// João: intimações com mais de 3 dias são irrelevantes

const JANELA_DIAS = parseInt(process.env.JANELA_DIAS || '3');
const cutoffDate = new Date();
cutoffDate.setDate(cutoffDate.getDate() - JANELA_DIAS);

// Após parsear as linhas do XLS, filtrar:
const filtradosPorData = processos.filter(p => {
  if (!p.dataEvento) return true; // sem data → inclui (conservador)
  const dataEvento = new Date(p.dataEvento);
  return dataEvento >= cutoffDate;
});

console.log(`[planilha] Filtro temporal: ${processos.length} total → ${filtradosPorData.length} nos últimos ${JANELA_DIAS} dias`);
return filtradosPorData;

Fix C — Deduplicação por signature_hash

-- Migration Supabase (executar uma vez):
ALTER TABLE processos_movimentacoes
  ADD COLUMN IF NOT EXISTS signature_hash text,
  ADD COLUMN IF NOT EXISTS fonte text DEFAULT 'cecilia_v2';

CREATE UNIQUE INDEX IF NOT EXISTS idx_processos_mov_signature
  ON processos_movimentacoes (organizacao_id, signature_hash)
  WHERE signature_hash IS NOT NULL;

-- No orquestrador (substituir UPDATE+INSERT fallback):
import { createHash } from 'crypto';

function gerarSignatureHash(cnj, evento, dataEvento) {
  return createHash('sha256')
    .update(`${cnj}|${evento}|${dataEvento || ''}`)
    .digest('hex')
    .slice(0, 32);
}

// Ao persistir resultado:
const hash = gerarSignatureHash(r.cnj, r.evento, r.dataEvento);
await supabase.from('processos_movimentacoes').upsert({
  organizacao_id: ORG_ID,
  signature_hash: hash,
  // ... demais campos
}, { onConflict: 'organizacao_id,signature_hash', ignoreDuplicates: false });
