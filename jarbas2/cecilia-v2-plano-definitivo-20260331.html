// Nova função no skill_orquestrador.mjs
async function enriquecerComMdFlow(processosXLS) {
  const cnjs = processosXLS.map(p => p.cnj).filter(Boolean);
  if (!cnjs.length) return processosXLS;

  const { data, error } = await supabase
    .from('processos')
    .select('id, numero_cnj, empresa_nome, reu_nome, polo_representado, status, ajg_concedida, fase_processual')

  return processosXLS.map(p => ({ ...p, _mdflow: mapa[p.cnj] || null }));
}

M3 Verificar tarefa aberta recente → ruído (Regra 3) skill_orquestrador.mjs

Se já existe tarefa aberta sobre o mesmo processo nos últimos 5 dias úteis, a nova intimação é ruído. Query na tabela atividades por processo_id + status não concluído + created_at recente.

// Nova função no skill_orquestrador.mjs
async function filtrarComTarefaAberta(processos) {
  const processoIds = processos.map(p => p._mdflow?.id).filter(Boolean);
  if (!processoIds.length) return processos;

  const cincoUteis = new Date();
  cincoUteis.setDate(cincoUteis.getDate() - 7); // ~5 dias úteis

  const { data } = await supabase
    .from('atividades')
    .select('processo_id')
    .in('processo_id', processoIds)
    .not('status', 'in', '("concluida","cancelada")')
    .is('deleted_at', null)
    .gte('created_at', cincoUteis.toISOString());

  const comTarefaAberta = new Set((data || []).map(a => a.processo_id));

  return processos.filter(proc => {
    const id = proc._mdflow?.id;
    if (id && comTarefaAberta.has(id)) {
      console.error(`[R3] Tarefa aberta recente — ruído: ${proc.cnj}`);
      return false;
    }
    return true;
  });
}

// Nova função de roteamento (pode ficar em eproc-utils.mjs ou skill_analisar_ia.mjs)
const TIPO_ATIVIDADE_POR_CLASSE = {
  'C': 'cobranca_documentos',
  'D': 'replica',            // ou manifestacao — depende do evento
  'E': 'sentenca',           // ou estrategica
  'F': 'validacao',          // João Victor sempre
  'G': 'supervisao',
};

async function rotearResponsavel(empresaNome, classe) {
  const tipoAtividade = TIPO_ATIVIDADE_POR_CLASSE[classe];
  if (!tipoAtividade) return null;

  // Normalizar empresa para tipo_reu
  const tipoReu = normalizarTipoReu(empresaNome);
  // normalizarTipoReu: "Boa Vista" → "Boa Vista" | "Santander" → "bancos" | etc.

  const { data } = await supabase
    .from('cecilia_equipe')
    .select('nome, prazo_dias')
    .contains('tipo_atividade', [tipoAtividade])
    .eq('ativo', true)
    .or(`tipo_reu.cs.{"${tipoReu}"},tipo_reu.cs.{"*"}`)
    .limit(1)
    .maybeSingle();

  return data || null;
}

// Exemplos de normalizarTipoReu():
// "Boa Vista SCPC" → "Boa Vista"
// "Banco Santander" → "bancos"  
// "Nubank" → "Nubank"
// "Realize" → "sessao_credito"
// Tudo mais → "*" (Heloísa pega improcedentes genéricos)

// Após classificação IA, enriquecer resultado:
if (['C','D','E','F'].includes(resultado.classe)) {
  const responsavel = await rotearResponsavel(
    proc._mdflow?.empresa_nome || proc._mdflow?.reu_nome,
    resultado.classe
  );

    resultado.responsavel = responsavel.nome;
    resultado.prazo_dias = resultado.prazo_dias || responsavel.prazo_dias;
  }
}

// Mapa de classe → prioridade
const PRIORIDADE_POR_CLASSE = { C: 'normal', D: 'normal', E: 'alta', F: 'urgente' };

// Nova função createAtividadeMdFlow (só se _mdflow.id existir)
async function criarAtividadeMdFlow(proc, resultado, responsavelId) {
  if (!['C','D','E','F'].includes(resultado.classe)) return null;
  if (!proc._mdflow?.id) return null; // sem processo no MdFlow

  const titulos = {
    C: `📋 Docs necessários — ${proc._mdflow.empresa_nome || proc._mdflow.reu_nome}`,
    D: `⚖️ ${resultado.acao || 'Providência jurídica'} — ${proc._mdflow.empresa_nome || proc._mdflow.reu_nome}`,
    E: `🚨 Estratégico — ${proc._mdflow.empresa_nome || proc._mdflow.reu_nome}`,
    F: `💰 VALORES — ${proc._mdflow.empresa_nome || proc._mdflow.reu_nome}`,
  };

  const descricao = [
    `[Cecília v2] ${resultado.justificativa || resultado.acao}`,
    `Evento: ${proc.ultimoEvento}`,
    `Classe: ${resultado.classe} | Confiança: ${Math.round((resultado.confianca || 0.5) * 100)}%`,
    resultado.prazo_dias ? `Prazo sugerido: ${resultado.prazo_dias} dias` : null,
  ].filter(Boolean).join('\n');

  const dataVencimento = resultado.prazo_dias
    ? new Date(Date.now() + resultado.prazo_dias * 86400000).toISOString().split('T')[0]
    : null;

  const { data, error } = await supabase.from('atividades').insert({
    organizacao_id: ORG_ID,
    processo_id: proc._mdflow.id,
    titulo: titulos[resultado.classe],
    descricao,
    status: 'aberta',
    prioridade: PRIORIDADE_POR_CLASSE[resultado.classe],
    responsavel_id: responsavelId || null, // buscar via cecilia_equipe.responsavel_id
    data_vencimento: dataVencimento,
  }).select('id').single();

  if (!error) {
    console.log(`[M7] Atividade criada: ${titulos[resultado.classe]} (${proc.cnj})`);
  }
  return data?.id || null;
}
