🔴 Root cause do BUG-01 — a confusão F vs G

O detectColumnIndices() busca a string "ultimo evento" / "evento" / "movimentacao" no cabeçalho. Quando o TJSC usa um nome de coluna não mapeado (ex: "Movimentação" com acento), a detecção falha e o fallback usa DEFAULT_INDICES = { ..., assunto: 5, evento: 6, ... }.

O problema é que quando a detecção falha de forma silenciosa, o sistema usa índice 5 (assunto) pensando que é 6 (evento). Precisamos de 3 fixes simultâneos:

3. Validação do evento extraídoSe col 6 parece assunto (contém vírgulas + "DIREITO"), tentar col 5 e 4 como fallback

// Wrapper seguro para page.evaluate:
async function safeEvaluate(page, fn, args, retries = 2) {
  for (let attempt = 0; attempt <= retries; attempt++) {
    if (page.isClosed()) throw new Error('page.isClosed() — não pode evaluate');
    try {
      await page.waitForLoadState('domcontentloaded', { timeout: 5000 }).catch(() => {});
      return await page.evaluate(fn, args);
    } catch (e) {
      if (attempt === retries) throw e;
      console.warn(`   ⚠️ evaluate tentativa ${attempt+1}/${retries}: ${e.message.slice(0,60)}`);
      await new Promise(r => setTimeout(r, 800 * (attempt + 1)));
    }
  }
}

// Chamada no orquestrador: if (needsEnriquecimento) await skillEnriquecerProcesso(page, cnj, processoId)

export async function skillEnriquecerProcesso(page, cnj, processoId) {
  if (!processoId) return; // sem processo_id, não atualiza

  const dados = await safeEvaluate(page, () => {
    const getText = (sels) => {
      for (const sel of sels) {
        const el = document.querySelector(sel);
        if (el?.textContent?.trim()) return el.textContent.trim();
      }
      return null;
    };
    return {
      // Órgão julgador / vara (seletores a confirmar com processo 5108947-30.2025.8.21.0001):
      orgaoJulgador: getText(['#fldOrgaoJulgador', '.orgaoJulgador', 'td.infraLabelObrigatorio + td']),
      // Juiz
      magistrado: getText(['.magistrado', '#fldMagistrado', '[id*="magistrado"]']),
      // Data distribuição
      dataDistribuicao: getText(['#fldDataAutuacao', '[id*="autuacao"]', '[id*="distribuicao"]']),
    };
  }).catch(() => null);

  if (!dados) return;

  const patch = {};
  if (dados.orgaoJulgador) {
    patch.vara_distribuicao = dados.orgaoJulgador.slice(0, 200);
    // Tentar vincular comarca_vara_id
    const { data: vara } = await supabase.from('comarcas_varas')
      .select('id').eq('vara', dados.orgaoJulgador).eq('organizacao_id', ORG_ID).maybeSingle();
    if (vara) patch.comarca_vara_id = vara.id;
  }
  if (dados.magistrado) {
    // Upsert em julgadores
    const { data: juiz } = await supabase.from('julgadores')
      .upsert({ nome: dados.magistrado, organizacao_id: ORG_ID, ativo: true },
               { onConflict: 'nome,organizacao_id' }).select('id').maybeSingle();
    if (juiz) patch.julgador_id = juiz.id;
  }

  if (Object.keys(patch).length > 0) {
    patch.enriquecido_em = new Date().toISOString();
    patch.enriquecido_por = 'cecilia_v2';
    await supabase.from('processos').update(patch).eq('id', processoId);
  }
}
