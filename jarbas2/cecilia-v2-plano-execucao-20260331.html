// Problema: um único erro de JSON anula o batch inteiro.
// Fix 1: tentar extrair JSON parcial do texto antes de desistir
function parseJsonArrayRobusto(texto) {
  // Tentativa 1: parse direto
  try {
    const parsed = JSON.parse(texto);
    if (Array.isArray(parsed)) return parsed;
  } catch {}

  // Tentativa 2: extrair array do texto (o modelo às vezes adiciona texto antes/depois)
  const match = texto.match(/\[[\s\S]*\]/);
  if (match) {
    try {
      const parsed = JSON.parse(match[0]);
      if (Array.isArray(parsed)) return parsed;
    } catch {}
  }

  // Tentativa 3: extrair objetos individuais
  const objetos = [...texto.matchAll(/\{[^{}]+\}/g)];
  if (objetos.length > 0) {
    return objetos.map(m => { try { return JSON.parse(m[0]); } catch { return null; } }).filter(Boolean);
  }

  return []; // falhou de verdade
}

// Wrapper seguro para page.evaluate()
async function safeEvaluate(page, fn, args, retries = 2) {
  for (let i = 0; i <= retries; i++) {
    if (page.isClosed()) throw new Error('page.isClosed() antes de evaluate');
    try {
      await page.waitForLoadState('domcontentloaded', { timeout: 8000 }).catch(() => {});
      return await page.evaluate(fn, args);
    } catch (e) {
      if (i === retries) throw e;
      console.log(`   ⚠️ evaluate falhou (tentativa ${i+1}): ${e.message.slice(0,60)} — retry...`);
      await new Promise(r => setTimeout(r, 800));
    }
  }
}

  const digits = cnj.replace(/\D/g, '');
  if (digits.length === 20) {

❓ Dúvida Resolvida: Como Funciona a Inserção?

Fluxo real do sistema (por processo)

Christian perguntou: "ele primeiro alimenta o controller e depois enriquece, ou faz tudo junto?"

Resposta: É tudo em uma passagem, processo por processo:

Acessa processo no EPROC → extrai eventos do DOM
Lê documentos (decisões, despachos) → texto extraído
Chama Haiku em batch → classifica evento + texto
Se D/E/F → chama Sonnet para análise profunda
Insere em intimacoes com tudo já preenchido
Controller mostra após a inserção ser concluída

O que você viu de "complementação posterior" foi o batchLookupProcessoIds — ele busca o processo_id para vários CNJs de uma vez antes de processar. Pode ter rodado e preenchido os campos NULL de processos que já estavam no banco. Não é inserção em duas etapas — é lookup batch que ocorreu em momento diferente.

⚠️ Ponto importante: Se pausou no meio, os processos já inseridos ficam no banco. Os não inseridos simplesmente não existem. Não tem estado intermediário corrompido.
