1. Número do Processo — Extrair 25 dígitos do XLS
skill_processar_planilha.mjs — modificação simples no parsing do CNJ

Christian pediu: 25 primeiros dígitos da esquerda. O CNJ brasileiro tem 20 dígitos no padrão NNNNNNN-DD.AAAA.J.TT.OOOO = 20 dígitos numéricos. O campo no XLS às vezes tem sufixos (nome da vara, texto extra) concatenados. A regex /\d{7}-\d{2}\.\d{4}\.\d{1}\.\d{2}\.\d{4}/ já existe e já funciona — mas se o EPROC às vezes usa formato diferente, extrair os primeiros 25 chars e aplicar regex é mais robusto.

Fix — skill_processar_planilha.mjs
function normalizarCNJ(raw) {
  const s = String(raw || '').trim();
  // Tentativa 1: regex padrão CNJ
  const m = s.match(/\d{7}-\d{2}\.\d{4}\.\d{1}\.\d{2}\.\d{4}/);
  if (m) return m[0];
  // Tentativa 2: primeiros 25 chars se parecer número
  const first25 = s.slice(0, 25).replace(/[^\d.\-]/g, '');
  const m2 = first25.match(/\d{7}-\d{2}\.\d{4}\.\d{1}\.\d{2}\.\d{4}/);
  if (m2) return m2[0];
  // Tentativa 3: só os 20 dígitos numéricos
  const digits = s.replace(/\D/g, '').slice(0, 20);
  if (digits.length === 20) {
    return `${digits.slice(0,7)}-${digits.slice(7,9)}.${digits.slice(9,13)}.${digits.slice(13,14)}.${digits.slice(14,16)}.${digits.slice(16)}`;
  }
  return s; // fallback: retorna como está
}

// Após capturar metadados, atualizar processos no Supabase:
async function enriquecerProcesso(cnj, metadados, processoId) {
  if (!processoId || !metadados) return;

  const updates = {};

  // Juiz
  if (metadados.juiz_nome) {
    // Upsert em julgadores
    const { data: juiz } = await supabase
      .from('julgadores')
      .upsert({ nome: metadados.juiz_nome, organizacao_id: ORG_ID, ativo: true },
               { onConflict: 'nome,organizacao_id' })
      .select('id').single();

    if (juiz?.id) updates.julgador_id = juiz.id;
  }

  // Vara + Comarca
  if (metadados.vara_nome) {
    const { data: vara } = await supabase
      .from('comarcas_varas')
      .select('id')
      .eq('vara', metadados.vara_nome)
      .eq('organizacao_id', ORG_ID)
      .single();

    if (vara?.id) {
      updates.comarca_vara_id = vara.id;
    } else {
      updates.vara_distribuicao = metadados.vara_nome;
    }
  }

  if (Object.keys(updates).length > 0) {
    updates.enriquecido_em = new Date().toISOString();
    updates.enriquecido_por = 'cecilia_v2';
    await supabase.from('processos').update(updates).eq('id', processoId);
  }
}

const OABs_ESCRITORIO = ['116571', '131163', '74025', '75466', '535101'];

function detectarPolo(advogadosAtivos, advogadosPassivos) {
  // Verificar se algum dos nossos OABs está no polo ativo
  const oabNosAtivos = advogadosAtivos.some(adv =>
    OABs_ESCRITORIO.some(oab => adv.oab?.includes(oab))
  );
  if (oabNosAtivos) return 'autor';

  const oabNosPassivos = advogadosPassivos.some(adv =>
    OABs_ESCRITORIO.some(oab => adv.oab?.includes(oab))
  );
  if (oabNosPassivos) return 'reu';

  return null; // não conseguiu detectar
}
