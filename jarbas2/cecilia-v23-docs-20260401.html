withRetry() — BUG-07 fix (retry com backoff) 
async function withRetry(fn, label, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try { return await fn(); }
    catch (e) {
      if (i === maxRetries - 1) throw e;
      await new Promise(r => setTimeout(r, 2000 * (i + 1)));
    }
  }
}

// Antes (v2.2): JSON.parse() — retorna [] em erro (contamina batch)
// Depois (v2.3): 3-stage fallback

function parseJsonArrayRobusto(text) {
  // Stage 1: Parse como array completo
  try { return JSON.parse(text); } catch {}

  // Stage 2: Extrair objetos individuais por regex
  const objects = text.match(/\{[^{}]*(?:\{[^{}]*\}[^{}]*)*\}/g);
  if (objects?.length) return objects.map(o => JSON.parse(o));

  // Stage 3: Parse como objeto unico
  try { return [JSON.parse(text)]; } catch {}

  return null; // NOT [] — caller enters fallback path
}

// NUNCA sobrescreve dados existentes — so preenche campos vazios
async function enrichProcess(cnj, xlsData, eprocMetadata) {
  const { data: proc } = await supabase
    .from('processos').select('*')
    .eq('numero_cnj', cnj).single();

  if (!proc) return;

  const updates = {};
  if (!proc.cliente_nome && xlsData?.autores)
    updates.cliente_nome = xlsData.autores;
  if (!proc.vara_distribuicao && eprocMetadata?.orgaoJulgador)
    updates.vara_distribuicao = eprocMetadata.orgaoJulgador;
  // ... etc para cada campo

  if (Object.keys(updates).length === 0) return;
  await supabase.from('processos').update(updates).eq('id', proc.id);
}

🔗 8. Deduplicacao Dual

evento_eproc (novo v2.3) Numero do evento no EPROC (ex: Evento 33). Verifica (processo_id, evento_eproc) unico. Quando evento_eproc extraido do DOM (100% no E2E)

// 1. signature_hash dedup

const { data: existing } = await supabase
  .from('intimacoes')
  .select('id')
  .eq('signature_hash', hash)

if (existing) { duplicatas++; continue; }

// 2. evento_eproc dedup (quando disponivel)
if (eventoEproc && processoId) {

    .select('id')
    .eq('processo_id', processoId)
    .eq('evento_eproc', eventoEproc)
