3.3 Layer 2: Storage (Supabase + Triggers)

Dual-Write Strategy

Cada movimento detectado gera INSERT em duas tabelas:

  1. processos_movimentacoes — Registro historico completo (330K+ rows). Imutavel, append-only.
  2. intimacoes — Fila de triagem para Controller. Classe A filtrada (nunca INSERT). Trigger pg_net dispara Edge Function.

Deduplicacao Dual

MecanismoCampoLogica
Primariosignature_hashSHA256(CNJ|evento|dataEvento), truncado 64 chars. CHECK antes INSERT.
Secundarioevento_eprocID numerico do evento no DOM EPROC. UNIQUE(processo_id, evento_eproc).

Trigger pg_net

-- Trigger: ao inserir em intimacoes, invocar Edge Function
CREATE OR REPLACE FUNCTION fn_trigger_analyze_intimacao()
RETURNS TRIGGER AS $$
BEGIN
  PERFORM net.http_post(
    url := current_setting('app.settings.edge_function_url') || '/analyze-intimacao',
    headers := jsonb_build_object(
      'Content-Type', 'application/json',
      'Authorization', 'Bearer ' || current_setting('app.settings.service_role_key')
    ),
    body := jsonb_build_object(
      'intimacao_id', NEW.id,
      'conteudo', NEW.conteudo,
      'tipo_evento', NEW.tipo_evento,
      'cnj', NEW.cnj,
      'tribunal', NEW.tribunal
    )
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER trg_analyze_intimacao
AFTER INSERT ON intimacoes
FOR EACH ROW
WHEN (NEW.classe_cecilia IS NULL AND NEW.fonte = 'cecilia_v3')
EXECUTE FUNCTION fn_trigger_analyze_intimacao();

11.3 rpc_controller_batch_update_status

CREATE OR REPLACE FUNCTION rpc_controller_batch_update_status(
  p_ids UUID[],
  p_status TEXT,
  p_user_id UUID
)
RETURNS JSONB
LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE affected INTEGER;
BEGIN
  IF p_status NOT IN ('nao_lida', 'lida', 'finalizada') THEN
    RETURN jsonb_build_object('success', false, 'error', 'invalid_status');
  END IF;
  IF array_length(p_ids, 1) > 100 THEN
    RETURN jsonb_build_object('success', false, 'error', 'batch_too_large');
  END IF;

  UPDATE intimacoes SET
    status = p_status,
    concluida_em = CASE WHEN p_status = 'finalizada' THEN now() ELSE concluida_em END,
    concluida_por_id = CASE WHEN p_status = 'finalizada' THEN p_user_id ELSE concluida_por_id END,
    updated_at = now()
  WHERE id = ANY(p_ids)
    AND organizacao_id = public.get_org_id();

  GET DIAGNOSTICS affected = ROW_COUNT;
  RETURN jsonb_build_object('success', true, 'affected', affected);
END;
$$;

CREATE OR REPLACE FUNCTION rpc_controller_batch_update_status(
  p_ids UUID[],
  p_status TEXT,
  p_user_id UUID DEFAULT NULL
) RETURNS JSONB
LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
  v_count INT;
BEGIN
  -- Validar status
  IF p_status NOT IN ('nao_lida','lida','finalizada','erro','transferida') THEN
    RETURN jsonb_build_object('success', false, 'message', 'Status invalido: ' || p_status);
  END IF;

  UPDATE intimacoes
  SET status = p_status,
      updated_at = now(),
      processado_por = COALESCE(p_user_id::text, processado_por),
      processado_em = CASE WHEN p_status = 'finalizada' THEN now() ELSE processado_em END,
      concluida_em = CASE WHEN p_status = 'finalizada' THEN now() ELSE concluida_em END,
      concluida_por_id = CASE WHEN p_status = 'finalizada' THEN p_user_id ELSE concluida_por_id END
  WHERE id = ANY(p_ids)
    AND deleted_at IS NULL;

  GET DIAGNOSTICS v_count = ROW_COUNT;

  RETURN jsonb_build_object(
    'success', true,
    'updated', v_count,
    'status', p_status
  );
END;
$$;

9. UPDATE intimacoes SET classe_cecilia, confianca_ia, analise_ia, ia_status='concluido', ...
10. Se classe E ou F → POST Telegram (D6)
