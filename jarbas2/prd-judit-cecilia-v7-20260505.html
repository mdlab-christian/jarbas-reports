validacao-w5.sql 
-- Rodar ao final do W5 (apos 7 dias de Judit ativo)
SELECT
  tipo_saida,
  COUNT(*)                                           AS total,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (), 1) AS pct
FROM intimacoes
WHERE created_at >= NOW() - INTERVAL '7 days'
  AND deleted_at IS NULL
GROUP BY tipo_saida
ORDER BY total DESC;
