CREATE OR REPLACE FUNCTION fn_mdzap_conversas(p_org_id UUID, p_numero_id UUID DEFAULT NULL)
RETURNS TABLE (id UUID, contato_numero TEXT, contato_nome TEXT, status TEXT, nao_lidas INT,
  ultimo_msg_at TIMESTAMPTZ, metadata JSONB, created_at TIMESTAMPTZ,
  numero_display TEXT, numero_nome TEXT, numero_proposito TEXT,
  lead_id UUID, lead_nome TEXT, lead_coluna TEXT, lead_awaiting_human BOOLEAN,
  cliente_id UUID, cliente_nome TEXT)
LANGUAGE sql SECURITY DEFINER SET search_path = public AS $$
  SELECT wc.id, wc.contato_numero, wc.contato_nome, wc.status, wc.nao_lidas,
    wc.ultimo_msg_at, wc.metadata, wc.created_at,
    wn.numero_e164, wn.nome, wn.proposito,
    l.id, l.nome, l.coluna, l.awaiting_human,
    cl.id, cl.nome
  FROM whatsapp_conversas wc
  JOIN whatsapp_numeros wn ON wn.id = wc.numero_id
  LEFT JOIN crm_leads l ON l.id = wc.lead_id
  LEFT JOIN clientes cl ON cl.id = wc.cliente_id
  WHERE wc.organizacao_id = p_org_id
    AND (p_numero_id IS NULL OR wc.numero_id = p_numero_id)
  ORDER BY wc.ultimo_msg_at DESC NULLS LAST LIMIT 200;
$$;
GRANT EXECUTE ON FUNCTION fn_mdzap_conversas(UUID, UUID) TO authenticated;
GRANT EXECUTE ON FUNCTION fn_mdzap_conversas(UUID, UUID) TO service_role;
