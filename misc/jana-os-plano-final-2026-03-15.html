S2
Nova RPC: jarbas_consume_tarefa_host P1
CREATE OR REPLACE FUNCTION jarbas_consume_tarefa_host(
  p_organizacao_id UUID, p_instance_id TEXT, p_host TEXT
) RETURNS SETOF jarbas_tarefas LANGUAGE plpgsql AS $$
DECLARE v jarbas_tarefas;
BEGIN
  SELECT * INTO v FROM jarbas_tarefas
  WHERE organizacao_id = p_organizacao_id
    AND status = 'pendente'
    AND (preferred_host = 'any' OR preferred_host = p_host)
    AND (scheduled_at IS NULL OR scheduled_at <= NOW())
  ORDER BY prioridade DESC, priority_order ASC, created_at ASC
  LIMIT 1 FOR UPDATE SKIP LOCKED;

  IF v.id IS NULL THEN RETURN; END IF;

  UPDATE jarbas_tarefas SET
    status = 'em_andamento', iniciado_em = NOW(),
    instance_id = p_instance_id, executor = p_host, executing_host = p_host
  WHERE id = v.id;
  v.status := 'em_andamento';
  RETURN NEXT v;
END; $$;

⚙️ Mudanças no Código

C1
task-consumer.mjs: concurrency=1 → 4 workers paralelos P1 · 30min · resolve o gargalo
const MAX_CONCURRENT = 4;
const HOST = process.env.JARBAS_HOST || 'macmini';

async function runWorkerPool() {
  console.log(`[task-consumer] 🚀 ${MAX_CONCURRENT} workers · host=${HOST}`);
  await Promise.all(
    Array.from({ length: MAX_CONCURRENT }, (_, i) => runWorker(i + 1))
  );
}

async function runWorker(id) {
  while (true) {
    try {
      const { data } = await supabase.rpc('jarbas_consume_tarefa_host', {
        p_organizacao_id: ORG_ID,
        p_instance_id: `${INSTANCE_ID}-w${id}`,
        p_host: HOST
      });
      const tarefa = Array.isArray(data) ? data[0] : data;
      if (!tarefa?.id) { await sleep(POLL_INTERVAL); continue; }
      await executarTarefa(tarefa);
      await sleep(INTER_TASK_DELAY);
    } catch (e) {
      console.error(`[worker-${id}] Erro:`, e.message);
      await sleep(POLL_INTERVAL * 2);
    }
  }
}
