Junction table modelo ↔ tags — sem isso Smart Blocks não funciona
CREATE TABLE IF NOT EXISTS olivia_modelo_tags (
  id             UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  organizacao_id UUID NOT NULL REFERENCES organizacoes(id) ON DELETE CASCADE,
  modelo_id      UUID NOT NULL REFERENCES olivia_modelos_peticao(id) ON DELETE CASCADE,
  tag_id         UUID NOT NULL REFERENCES olivia_tags(id) ON DELETE CASCADE,
  obrigatoria    BOOLEAN DEFAULT FALSE,
  ordem          INTEGER DEFAULT 0,
  created_at     TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(organizacao_id, modelo_id, tag_id)
);
CREATE INDEX IF NOT EXISTS idx_modelo_tags_modelo ON olivia_modelo_tags(modelo_id);
CREATE INDEX IF NOT EXISTS idx_modelo_tags_tag ON olivia_modelo_tags(tag_id);
ALTER TABLE olivia_modelo_tags ENABLE ROW LEVEL SECURITY;

Hook useSmartBlocksData — lógica de filtro

// 1. Buscar tags vinculadas ao modelo selecionado
const { data: modeloTags } = useQuery({
  queryKey: ['olivia-modelo-tags', modeloId],
  queryFn: () => supabase
    .from('olivia_modelo_tags')
    .select('tag_id, obrigatoria, ordem, olivia_tags(id,nome,secao_id, olivia_tags_secao(id,nome,ordem))')
    .eq('modelo_id', modeloId)
    .eq('organizacao_id', organizacao_id)
    .order('ordem'),
  staleTime: 30000,
});

// 2. Buscar blocos filtrados pelas tags do modelo
const tagIds = modeloTags?.map(mt => mt.tag_id) ?? [];
const { data: blocos } = useQuery({
  queryKey: ['olivia-blocos-modelo', modeloId],
  queryFn: () => supabase
    .from('olivia_blocos_texto')
    .select('id, titulo, conteudo, tag_ids')
    .eq('organizacao_id', organizacao_id)
    .is('deleted_at', null)
    .overlaps('tag_ids', tagIds),  // tag_ids && tagIds
  enabled: tagIds.length > 0,
  staleTime: 30000,
});
