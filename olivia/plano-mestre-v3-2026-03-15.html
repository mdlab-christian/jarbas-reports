const payload = {

  tags_substituicao: {
    // Da tabela processos + dados do processo

    reu_nome: processo.empresas?.nome,
    numero_cnj: processo.numero_cnj,
    advogado_nome: advogado.nome,
    advogado_oab: advogado.oab,
    valor_causa: processo.valor_causa,
    tribunal: processo.tribunal,
    cidade: processo.cidade || 'Porto Alegre',
    data_hoje: new Date().toLocaleDateString('pt-BR'),
  },
  config: {
    fonte: 'Arial',
    cabecalho_escritorio: 'MP ADVOCACIA\nOAB/RS XXXXX',
    rodape_texto: 'Documento gerado por OlivIA — MdFlow',
  }
};

const response = await fetch(
  'https://script.google.com/macros/s/AKfycbyNw1hVqL0VfSodWSDWSW3Tl8btsIrsnLQl5nrVkPPZJBfHolxnA-gpjeN5mmq7wVuZIg/exec',
  { method: 'POST', headers: {'Content-Type':'application/json'}, body: JSON.stringify(payload) }
);
const result = await response.json();

if (result.success) {
  // Salvar em olivia_historico

    organizacao_id, processo_id: processo.id, modo: 'SMART_BLOCKS',

    google_doc_id: result.google_doc_id,
    status: 'SUCESSO',
    meta: { blocos_ids, titulo: payload.titulo_documento }
  });
  setState(prev => ({ ...prev, google_doc_url: result.google_doc_url, status_geracao: 'SUCESSO' }));
}

Mecanismo Completo — Triagem

O StepTriagem tem dois motores: motor='n8n' (Express) e motor='jarbas' (Agente). Para Express, o fluxo correto é:

// StepTriagem (motor='n8n') — fluxo correto

// 1. Detectar se análise é obrigatória
const temTags = await supabase
  .from('olivia_modelo_tags')
  .select('id')
  .eq('modelo_id', modeloGrupoId)
  .limit(1);
const analiseObrigatoria = temTags.data?.length > 0;

// 2. Se obrigatória e usuário fez upload → disparar OLIVIA-ANALISE
//    OLIVIA-ANALISE = workflow TupOpRfZXkTHzzEuBOk7f (mantido intacto)
if (analiseObrigatoria && provas_urls.length > 0) {
  await callWebhook('mdflow/olivia/analise', { processo_id, provas_urls, organizacao_id });
  // polling olivia_analises até status='CONCLUIDA'
}

// 3. Chamar WH-OLIVIA-TRIAGEM-V1 (novo — a criar no C2)
//    Path: mdflow/olivia/triagem
const triagem = await callWebhook('mdflow/olivia/triagem', {
  processo_id,
  modelo_grupo_id: modeloGrupoId,
  tipo_peticao_id: tipoPeticaoId,
  analise_id: analise?.id ?? null,
  analise_resultado: analise?.resultado ?? null,
  instrucoes_ia: modelo.instrucoes_ia ?? null,  // D12: contexto extra
  processo_context: {
    numero_processo: dadosProcesso.numero_cnj,
    reu_nome: dadosProcesso.empresas?.nome,
    cliente_nome: dadosProcesso.clientes?.nome,
    valor_causa: dadosProcesso.valor_causa,
    restricoes: dadosProcesso.restricoes,       // tabela processos é rica
    historico_processos: dadosProcesso.historico,
    // ... demais campos relevantes da tabela processos
  }
});
// Resultado: { perguntas: [{id, pergunta, tipo, obrigatorio, opcoes}] }
