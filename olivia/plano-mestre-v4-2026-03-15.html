Payload Apps Script — Completo

// SmartBlocksStep3Gerar — payload completo para Apps Script

// 1. Buscar cabecalho_escritorio do advogado selecionado
const { data: advogado } = await supabase
  .from('advogados')
  .select('nome, oab, oab_uf, cabecalho_escritorio')
  .eq('id', wizardState.advogado_id)
  .single();

const cabecalho = advogado.cabecalho_escritorio ||
  `${advogado.nome}\nOAB/${advogado.oab_uf} ${advogado.oab}`;

// 2. Montar payload
const payload = {
  titulo_documento: `Petição - ${processo.numero_cnj} - ${modelo.nome}`,
  google_folder_id: '1JTbKJqbmPzRTLlqRYZEZ5pNY-w8r1ALZ',
  secoes: blocosSelecionados.map((bloco, idx) => ({
    id: bloco.id,
    titulo: bloco.titulo,
    conteudo: bloco.conteudo,   // HTML → stripHtml no Apps Script
    tipo: idx === 0 ? 'header' : idx === blocos.length-1 ? 'conclusion' : 'body',
    ordem: idx
  })),
  tags_substituicao: {
    cliente_nome: processo.clientes?.nome,
    reu_nome: processo.empresas?.nome,
    numero_cnj: processo.numero_cnj,
    advogado_nome: advogado.nome,
    advogado_oab: `OAB/${advogado.oab_uf} ${advogado.oab}`,
    valor_causa: processo.valor_causa,
    tribunal: processo.tribunal,
    cidade: processo.cidade || 'Porto Alegre',
    data_hoje: new Date().toLocaleDateString('pt-BR'),
  },
  config: {
    fonte: 'Arial',
    cabecalho_escritorio: cabecalho,   // ← D14: vem de advogados.cabecalho_escritorio
    rodape_texto: 'Documento gerado por OlivIA — MdFlow',
  }
};

// 3. Chamar Apps Script diretamente (sem N8N)
const res = await fetch(APPS_SCRIPT_URL, {
  method: 'POST',
  headers: {'Content-Type': 'application/json'},
  body: JSON.stringify(payload)
});
const result = await res.json();

// 4. Salvar no historico
if (result.success) {
  await supabase.from('olivia_historico').insert({
    organizacao_id, processo_id: processo.id,
    modo: 'SMART_BLOCKS',           // ← D14: era 'MANUAL', agora correto
    google_doc_url: result.google_doc_url,
    google_doc_id: result.google_doc_id,
    status: 'SUCESSO',
    meta: { blocos_ids, titulo: payload.titulo_documento }
  });
}
