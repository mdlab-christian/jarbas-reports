// webhook-main.mjs — adicionar após /olivia/upload-documento
if (req.method === 'POST' && pathname === '/olivia/docx-converter') {
  try {
    const { historico_id, html_content, filename, organizacao_id } = await readBody(req);
    if (!historico_id || !html_content) return send(400, { ok: false, error: 'historico_id e html_content obrigatórios' });

    const { execSync } = await import('child_process');
    const { writeFileSync, readFileSync, unlinkSync } = await import('fs');
    const tmpHtml = `/tmp/peticao-${historico_id}.html`;
    const tmpDocx = `/tmp/peticao-${historico_id}.docx`;

    writeFileSync(tmpHtml, html_content, 'utf-8');
    execSync(`pandoc -f html -t docx --reference-doc=/Users/jarbas/jarbas/assets/olivia-template.docx -o ${tmpDocx} ${tmpHtml}`);

    const docxBuffer = readFileSync(tmpDocx);
    const storagePath = `${organizacao_id || ORG_ID}/${historico_id}/peticao.docx`;

    const { supabase } = await import('./lib/supabase.mjs');
    const { data: upload } = await supabase.storage.from('peticoes')
      .upload(storagePath, docxBuffer, { contentType: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document', upsert: true });

    const { data: { publicUrl } } = supabase.storage.from('peticoes').getPublicUrl(storagePath);

    await supabase.from('olivia_historico').update({ pdf_url: publicUrl, updated_at: new Date().toISOString() }).eq('id', historico_id);

    unlinkSync(tmpHtml); unlinkSync(tmpDocx);
    return send(200, { ok: true, pdf_url: publicUrl });
  } catch (e) { return send(500, { ok: false, error: e.message }); }
}

Regras de Tipografia e UX
  • Font família: Manrope (títulos, peso 800–900) + Inter (corpo, peso 400–600)
  • Hierarquia: Page title 1.8rem 900 · Section 1.2rem 800 · Label JetBrains Mono .7rem uppercase
  • Accent verde: #20c997 — apenas para: ações primárias, indicadores de sucesso, progresso
  • Feedback visual: Todo botão "Avançar" desabilitado deve ter tooltip explicando por quê
  • Loading states: Skeleton em vez de spinner para listas. Spinner para ações pontuais.
  • Mobile: Padding mínimo 16px. Cards em coluna única. Bottom sheet para modais.

Fluxo de Trabalho — JARBAS + Claude Code
Metodologia Pair-Programming — Loop por Conversa
1. JARBAS investiga → lê SPEC + PRD da conversa + codebase atual
2. JARBAS executa → edita arquivos diretamente (write/edit tools), faz queries SQL via Management API, chama N8N API, commita
3. Claude Code revisa → recebe SPEC + PRD + git diff → identifica gaps/drifts → aplica correções finais
4. JARBAS valida → roda E2E básico → confirma critério de sucesso
5. Repetir → se CC encontrou gaps, JARBAS corrige e CC revisa de novo até limpo
