import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

// Apps Script Web App URL
const APPS_SCRIPT_URL = `https://script.google.com/macros/s/AKfycbw...
  1korRdpdAn2y7kr-RRZQyPEoJ0XroqM__xhzSnkqWEpv38f6vBaiK_5HI/exec`;

// Input interface
interface SmartBlocksRequest {
  blocos_ids: string[];        // Array of olivia_blocos_texto.id
  modelo_id: string;          // olivia_modelos_peticao.id
  processo_id: string;        // processos.id
  org_id: string;             // organizacoes.id
  user_id?: string;           // Optional: from JWT
  titulo?: string;            // Optional: custom title
}

// Output interface
interface SmartBlocksResponse {
  success: boolean;
  doc_url: string | null;
  doc_id: string | null;
  historico_id: string | null;
  error?: string;
  blocos_merged: number;
  tokens_estimated: number;
}

serve(async (req: Request) => {
  // CORS headers
  if (req.method === 'OPTIONS') {
    return new Response('ok', {
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'POST, OPTIONS',
        'Access-Control-Allow-Headers': 'authorization, content-type, x-client-info',
      },
    });
  }

  try {
    // 1. Parse request
    const body: SmartBlocksRequest = await req.json();
    const { blocos_ids, modelo_id, processo_id, org_id, titulo } = body;

    // 2. Validate required fields
    if (!blocos_ids?.length || !modelo_id || !processo_id || !org_id) {
      return new Response(
        JSON.stringify({ 
          success: false, 
          error: 'Missing required fields: blocos_ids, modelo_id, processo_id, org_id' 
        }),
        { status: 400, headers: { 'Content-Type': 'application/json' } }
      );
    }

    // 3. Initialize Supabase client with service role
    const supabaseAdmin = createClient(
      Deno.env.get('SUPABASE_URL') ?? '',
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? ''
    );

    // 4. Fetch blocos with content - ORDER BY ordem ASC
    const { data: blocos, error: blocosError } = await supabaseAdmin
      .from('olivia_blocos_texto')
      .select('id, titulo, conteudo, tipo, ordem, tags')
      .in('id', blocos_ids)
      .eq('status', 'active')  // P0-A: status must be 'active' not 'published'
      .order('ordem', { ascending: true });

    if (blocosError || !blocos?.length) {
      throw new Error(`Failed to fetch blocos: ${blocosError?.message || 'No blocks found'}`);
    }

    // 5. Fetch processo context for placeholder replacement
    const { data: processo } = await supabaseAdmin
      .from('processos')
      .select(`
        id, 
        numero_cnj, 
        cliente_nome,
        empresa_re,
        valor_causa,
        orgao_id,
        orgaos:orgao_id (nome, sigla)
      `)
      .eq('id', processo_id)
      .single();

    // 6. Fetch modelo for title
    const { data: modelo } = await supabaseAdmin
      .from('olivia_modelos_peticao')
      .select('nome, tipo_peticao')
      .eq('id', modelo_id)
      .single();

    // 7. Merge blocos content with placeholder replacement
    const placeholders = {
      '{{CLIENTE_NOME}}': processo?.cliente_nome || '[CLIENTE]',
      '{{EMPRESA_RE}}': processo?.empresa_re || '[EMPRESA RÉ]',
      '{{NUMERO_CNJ}}': processo?.numero_cnj || '[CNJ]',
      '{{VALOR_CAUSA}}': processo?.valor_causa?.toLocaleString('pt-BR', { 
        style: 'currency', 
        currency: 'BRL' 
      }) || '[VALOR]',
      '{{ORGAO_NOME}}': processo?.orgaos?.nome || '[ÓRGÃO]',
      '{{DATA_ATUAL}}': new Date().toLocaleDateString('pt-BR'),
    };

    let mergedContent = blocos
      .map(b => b.conteudo)
      .join('\n\n<hr/>\n\n');

    // Replace all placeholders
    Object.entries(placeholders).forEach(([key, value]) => {
      mergedContent = mergedContent.replace(new RegExp(key, 'g'), value);
    });

    // 8. Call Apps Script to create Google Doc
    const appsScriptPayload = {
      action: 'createDocument',
      title: titulo || `${modelo?.nome || 'Petição'} - ${processo?.numero_cnj || processo_id}`,
      htmlContent: mergedContent,
      folderId: Deno.env.get('GOOGLE_DRIVE_FOLDER_ID'),
      shareWith: ['advogado@cliente.com'], // TODO: fetch from org settings
    };

    const appsScriptResponse = await fetch(APPS_SCRIPT_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(appsScriptPayload),
    });

    const appsScriptResult = await appsScriptResponse.json();

    if (!appsScriptResult.success) {
      throw new Error(`Apps Script error: ${appsScriptResult.error}`);
    }

    // 9. Create olivia_historico record
    const { data: historico, error: historicoError } = await supabaseAdmin
      .from('olivia_historico')
      .insert({
        processo_id,
        org_id,
        modelo_id,
        tipo_operacao: 'smart_blocks',
        status: 'SUCESSO',
        texto_gerado: mergedContent,
        google_doc_url: appsScriptResult.docUrl,
        google_doc_id: appsScriptResult.docId,
        meta: {
          blocos_ids,
          blocos_count: blocos.length,
          tokens_estimated: Math.ceil(mergedContent.length / 4),
          placeholders_replaced: Object.keys(placeholders).length,
        },
      })
      .select('id')
      .single();

    if (historicoError) {
      console.error('Failed to create historico:', historicoError);
    }

    // 10. Return success response
    return new Response(
      JSON.stringify({
        success: true,
        doc_url: appsScriptResult.docUrl,
        doc_id: appsScriptResult.docId,
        historico_id: historico?.id || null,
        blocos_merged: blocos.length,
        tokens_estimated: Math.ceil(mergedContent.length / 4),
      }),
      {
        status: 200,
        headers: {
          'Content-Type': 'application/json',
          'Access-Control-Allow-Origin': '*',
        },
      }
    );
  } catch (error) {
    console.error('Smart Blocks Error:', error);
    return new Response(
      JSON.stringify({
        success: false,
        error: error.message,
        doc_url: null,
        doc_id: null,
        historico_id: null,
      }),
      {
        status: 500,
        headers: {
          'Content-Type': 'application/json',
          'Access-Control-Allow-Origin': '*',
        },
      }
    );
  }
});

4.3 Google Apps Script: Document Generator

// Apps Script: Code.gs

function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    
    switch (payload.action) {
      case 'createDocument':
        return createDocument(payload);
      default:
        return ContentService.createTextOutput(
          JSON.stringify({ success: false, error: 'Unknown action' })
        ).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (error) {
    return ContentService.createTextOutput(
      JSON.stringify({ success: false, error: error.message })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

function createDocument(payload) {
  const { title, htmlContent, folderId, shareWith } = payload;
  
  // 1. Create new Google Doc
  const doc = DocumentApp.create(title);
  const body = doc.getBody();
  
  // 2. Parse HTML and insert content
  // Note: Google Docs has limited HTML support
  body.appendParagraph(title)
    .setHeading(DocumentApp.ParagraphHeading.HEADING1);
  
  // Split by HR and process each block
  const blocks = htmlContent.split(//gi);
  
  blocks.forEach((block, index) => {
    if (index > 0) {
      body.appendHorizontalRule();
    }
    
    // Strip HTML tags but preserve structure
    const text = block
      .replace(/<\/p>/gi, '\n')
      .replace(//gi, '\n')
      .replace(/<[^>]*>/g, '')
      .trim();
    
    if (text) {
      body.appendParagraph(text);
    }
  });
  
  doc.saveAndClose();
  
  // 3. Move to specified folder
  if (folderId) {
    const file = DriveApp.getFileById(doc.getId());
    const folder = DriveApp.getFolderById(folderId);
    folder.addFile(file);
    DriveApp.getRootFolder().removeFile(file);
  }
  
  // 4. Share with specified emails
  if (shareWith && shareWith.length) {
    shareWith.forEach(email => {
      DriveApp.getFileById(doc.getId())
        .addEditor(email);
    });
  }
  
  return ContentService.createTextOutput(
    JSON.stringify({
      success: true,
      docId: doc.getId(),
      docUrl: doc.getUrl(),
    })
  ).setMimeType(ContentService.MimeType.JSON);
}

4.4 Frontend: useSmartBlocks Hook

📁 File: src/hooks/olivia/useSmartBlocks.ts

import { useState, useCallback } from 'react';
import { supabase } from '@/lib/supabase';
import { toast } from 'sonner';

interface SmartBlocksParams {
  blocos_ids: string[];
  modelo_id: string;
  processo_id: string;
  org_id: string;
  titulo?: string;
}

interface SmartBlocksResult {
  success: boolean;
  doc_url: string | null;
  doc_id: string | null;
  historico_id: string | null;
  error?: string;
}

export function useSmartBlocks() {
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<SmartBlocksResult | null>(null);

  const generateDocument = useCallback(async (params: SmartBlocksParams) => {
    setLoading(true);
    setResult(null);

    try {
      // Validate inputs
      if (!params.blocos_ids?.length) {
        throw new Error('Selecione pelo menos um bloco');
      }
      if (!params.processo_id) {
        throw new Error('Processo não selecionado');
      }

      // Call Edge Function
      const { data, error } = await supabase.functions.invoke(
        'olivia-smart-blocks',
        {
          body: params,
        }
      );

      if (error) {
        throw error;
      }

      if (!data?.success) {
        throw new Error(data?.error || 'Falha ao gerar documento');
      }

      setResult(data);
      toast.success('Documento gerado com sucesso!', {
        description: `${params.blocos_ids.length} blocos mesclados`,
        action: {
          label: 'Abrir Doc',
          onClick: () => window.open(data.doc_url, '_blank'),
        },
      });

      return data;
    } catch (error) {
      const errorResult: SmartBlocksResult = {
        success: false,
        doc_url: null,
        doc_id: null,
        historico_id: null,
        error: error.message,
      };
      setResult(errorResult);
      toast.error('Erro ao gerar documento', {
        description: error.message,
      });
      return errorResult;
    } finally {
      setLoading(false);
    }
  }, []);

  return {
    generateDocument,
    loading,
    result,
  };
}
