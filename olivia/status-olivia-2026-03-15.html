SELECT p.*, c.nome, c.cpf_cnpj, c.estado_civil, c.profissao, e.razao_social, e.cnpj, a.nome, a.oab, a.oab_uf FROM processos p LEFT JOIN clientes c ON c.id=p.cliente_id LEFT JOIN empresas e ON e.id=p.empresa_id LEFT JOIN advogados a ON a.id=p.advogado_id WHERE p.id = historico.processo_id
